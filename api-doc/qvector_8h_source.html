37 explicit qvector(const std::vector<complex> &vector, bool validate)
38 : qudits(std::log2(vector.size())) {
39 if (Levels == 2) {
40 auto numElements = std::log2(vector.size());
41 if (std::floor(numElements) != numElements)
42 throw std::runtime_error(
43 "Invalid state vector passed to qvector initialization - number of "
44 "elements must be power of 2.");
45 }
46 if (validate) {
47 auto norm = std::inner_product(
48 vector.begin(), vector.end(), vector.begin(),
49 complex{0., 0.}, [](auto a, auto b) { return a + b; },
50 [](auto a, auto b) { return std::conj(a) * b; })
51 .real();
52 if (std::fabs(1.0 - norm) > 1e-4)
53 throw std::runtime_error("Invalid vector norm for qudit allocation.");
54 }
55 std::vector<QuditInfo> targets;
56 for (auto &q : qudits)
57 targets.emplace_back(QuditInfo{Levels, q.id()});
58
59 auto precision = std::is_same_v<complex::value_type, float>
60 ? simulation_precision::fp32
61 : simulation_precision::fp64;
62 getExecutionManager()->initializeState(targets, vector.data(), precision);
63 }
