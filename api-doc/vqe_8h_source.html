22template <typename QuantumKernel, typename... Args,
23 typename = std::enable_if_t<
24 std::is_invocable_v<QuantumKernel, std::vector<double>, Args...>>>
25static inline optimization_result
26remote_vqe(cudaq::quantum_platform &platform, QuantumKernel &&kernel,
27 cudaq::spin_op &H, cudaq::optimizer &optimizer,
28 cudaq::gradient *gradient, const int n_params,
29 const std::size_t shots, Args &&...args) {
30 auto ctx = std::make_unique<ExecutionContext>("observe", shots);
31 ctx->kernelName = cudaq::getKernelName(kernel);
32 ctx->spin = &H;
33 platform.set_exec_ctx(ctx.get());
34 auto serializedArgsBuffer = serializeArgs(args...);
35 platform.launchVQE(ctx->kernelName, serializedArgsBuffer.data(), gradient, H,
36 optimizer, n_params, shots);
37 platform.reset_exec_ctx();
38 return ctx->optResult.value_or(optimization_result{});
39}

93template <typename QuantumKernel, typename... Args,
94 typename = std::enable_if_t<
95 std::is_invocable_v<QuantumKernel, std::vector<double>, Args...>>>
96optimization_result vqe(QuantumKernel &&kernel, cudaq::spin_op H,
97 cudaq::optimizer &optimizer, const int n_params,
98 Args &&...args) {
99 static_assert(
100 std::is_invocable_v<QuantumKernel, std::vector<double>, Args...>,
101 "Invalid parameterized quantum kernel expression. Must have "
102 "void(std::vector<double>, <Args...>) signature, or provide "
103 "std::tuple<Args...>(std::vector<double>) ArgMapper function object.");
104 if (optimizer.requiresGradients()) {
105 throw std::invalid_argument("Provided cudaq::optimizer requires gradients. "
106 "Please provide a cudaq::gradient instance.");
107 }
108
109 auto &platform = cudaq::get_platform();
110 if (platform.get_remote_capabilities().vqe)
111 return __internal__::remote_vqe(platform, kernel, H, optimizer,
112 /*gradient=*/nullptr, n_params, /*shots=*/0,
113 args...);
114
115 return optimizer.optimize(n_params, [&](const std::vector<double> &x,
116 std::vector<double> &grad_vec) {
117 double e = cudaq::observe(kernel, H, x, args...);
118 return e;
119 });
120}

239template <typename QuantumKernel, typename... Args,
240 typename = std::enable_if_t<
241 std::is_invocable_v<QuantumKernel, std::vector<double>, Args...>>>
242optimization_result vqe(QuantumKernel &&kernel, cudaq::gradient &gradient,
243 cudaq::spin_op H, cudaq::optimizer &optimizer,
244 const int n_params, Args &&...args) {
245 static_assert(
246 std::is_invocable_v<QuantumKernel, std::vector<double>> ||
247 std::is_invocable_v<QuantumKernel, std::vector<double>, Args...>,
248 "Invalid parameterized quantum kernel expression. Must have "
249 "void(std::vector<double>, <Args...>) signature, or provide "
250 "std::tuple<Args...>(std::vector<double>) ArgMapper function object.");
251
252 auto &platform = cudaq::get_platform();
253 if (platform.get_remote_capabilities().vqe)
254 return __internal__::remote_vqe(platform, kernel, H, optimizer, &gradient,
255 n_params,
256 /*shots=*/0, args...);
257
258 auto requires_grad = optimizer.requiresGradients();
259 // If there are additional arguments, we need to clone the gradient and
260 // provide it the concrete arguments.
261 // Note: the strange initialization of newGrad is to avoid a C++17 compiler
262 // error that happens because the `swap` is ambiguous between the unique_ptr
263 // and the qubit swap.
264 std::unique_ptr<cudaq::gradient> newGrad = [&]() {
265 if (requires_grad) {
266 auto newGrad_ = gradient.clone();
267 if constexpr (sizeof...(args) > 0)
268 newGrad_->setArgs(kernel, args...);
269 return newGrad_;
270 }
271 return std::unique_ptr<cudaq::gradient>();
272 }();
273 return optimizer.optimize(n_params, [&](const std::vector<double> &x,
274 std::vector<double> &grad_vec) {
275 double e = cudaq::observe(kernel, H, x, args...);
276 if (requires_grad)
277 newGrad->compute(x, grad_vec, H, e);
278 return e;
279 });
280}
