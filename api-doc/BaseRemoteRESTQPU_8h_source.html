204 void setTargetBackend(const std::string &backend) override {
205 cudaq::info("Remote REST platform is targeting {}.", backend);
206
207 // First we see if the given backend has extra config params
208 auto mutableBackend = backend;
209 if (mutableBackend.find(";") != std::string::npos) {
210 auto split = cudaq::split(mutableBackend, ';');
211 mutableBackend = split[0];
212 // Must be key-value pairs, therefore an even number of values here
213 if ((split.size() - 1) % 2 != 0)
214 throw std::runtime_error(
215 "Backend config must be provided as key-value pairs: " +
216 std::to_string(split.size()));
217
218 // Add to the backend configuration map
219 for (std::size_t i = 1; i < split.size(); i += 2) {
220 // No need to decode trivial true/false values
221 if (split[i + 1].starts_with("base64_")) {
222 split[i + 1].erase(0, 7); // erase "base64_"
223 std::vector<char> decoded_vec;
224 if (auto err = llvm::decodeBase64(split[i + 1], decoded_vec))
225 throw std::runtime_error("DecodeBase64 error");
226 std::string decodedStr(decoded_vec.data(), decoded_vec.size());
227 cudaq::info("Decoded {} parameter from '{}' to '{}'", split[i],
228 split[i + 1], decodedStr);
229 backendConfig.insert({split[i], decodedStr});
230 } else {
231 backendConfig.insert({split[i], split[i + 1]});
232 }
233 }
234 }
235
236 // Turn on emulation mode if requested
237 auto iter = backendConfig.find("emulate");
238 emulate = iter != backendConfig.end() && iter->second == "true";
239
240 // Print the IR if requested
241 printIR = getEnvBool("CUDAQ_DUMP_JIT_IR", printIR);
242
243 // Get additional debug values
244 disableMLIRthreading =
245 getEnvBool("CUDAQ_MLIR_DISABLE_THREADING", disableMLIRthreading);
246 enablePrintMLIREachPass =
247 getEnvBool("CUDAQ_MLIR_PRINT_EACH_PASS", enablePrintMLIREachPass);
248
249 // If the very verbose enablePrintMLIREachPass flag is set, then
250 // multi-threading must be disabled.
251 if (enablePrintMLIREachPass) {
252 disableMLIRthreading = true;
253 }
254
258 std::string fileName = mutableBackend + std::string(".yml");
259 auto configFilePath = platformPath / fileName;
260 cudaq::info("Config file path = {}", configFilePath.string());
261 std::ifstream configFile(configFilePath.string());
262 std::string configYmlContents((std::istreambuf_iterator<char>(configFile)),
263 std::istreambuf_iterator<char>());
264 cudaq::config::TargetConfig config;
265 llvm::yaml::Input Input(configYmlContents.c_str());
266 Input >> config;
267 if (config.BackendConfig.has_value()) {
268 if (!config.BackendConfig->PlatformLoweringConfig.empty()) {
269 cudaq::info("Appending lowering pipeline: {}",
270 config.BackendConfig->PlatformLoweringConfig);
271 passPipelineConfig +=
272 "," + config.BackendConfig->PlatformLoweringConfig;
273 }
274 if (!config.BackendConfig->CodegenEmission.empty()) {
275 cudaq::info("Set codegen translation: {}",
276 config.BackendConfig->CodegenEmission);
277 codegenTranslation = config.BackendConfig->CodegenEmission;
278 }
279 if (!config.BackendConfig->PostCodeGenPasses.empty()) {
280 cudaq::info("Adding post-codegen lowering pipeline: {}",
281 config.BackendConfig->PostCodeGenPasses);
282 postCodeGenPasses = config.BackendConfig->PostCodeGenPasses;
283 }
284 }
285 std::string allowEarlyExitSetting =
286 (codegenTranslation == "qir-adaptive") ? "1" : "0";
287 passPipelineConfig = std::string("cc-loop-unroll{allow-early-exit=") +
288 allowEarlyExitSetting + "}," + passPipelineConfig;
289
290 auto disableQM = backendConfig.find("disable_qubit_mapping");
291 if (disableQM != backendConfig.end() && disableQM->second == "true") {
292 // Replace the qubit-mapping{device=<>} with
293 // qubit-mapping{device=bypass} to effectively disable the qubit-mapping
294 // pass. Use $1 - $4 to make sure any other pass options are left
295 // untouched.
296 std::regex qubitMapping(
297 "(.*)qubit-mapping\\{(.*)device=[^,\\}]+(.*)\\}(.*)");
298 std::string replacement("$1qubit-mapping{$2device=bypass$3}$4");
299 passPipelineConfig =
300 std::regex_replace(passPipelineConfig, qubitMapping, replacement);
301 cudaq::info("disable_qubit_mapping option found, so updated lowering "
302 "pipeline to {}",
303 passPipelineConfig);
304 }
305
306 // Set the qpu name
307 qpuName = mutableBackend;
308
309 // Create the ServerHelper for this QPU and give it the backend config
310 serverHelper = cudaq::registry::get<cudaq::ServerHelper>(qpuName);
311 serverHelper->initialize(backendConfig);
312 serverHelper->updatePassPipeline(platformPath, passPipelineConfig);
313
314 // Give the server helper to the executor
315 executor->setServerHelper(serverHelper.get());
316 }

319 nlohmann::json formOutputNames(const std::string &codegenTranslation,
320 const std::string &codeStr) {
321 // Form an output_names mapping from codeStr
322 nlohmann::json output_names;
323 std::vector<char> bitcode;
324 if (codegenTranslation.starts_with("qir")) {
325 // decodeBase64 will throw a runtime exception if it fails
326 if (llvm::decodeBase64(codeStr, bitcode)) {
327 cudaq::info("Could not decode codeStr {}", codeStr);
328 } else {
329 llvm::LLVMContext llvmContext;
330 auto buffer = llvm::MemoryBuffer::getMemBufferCopy(
331 llvm::StringRef(bitcode.data(), bitcode.size()));
332 auto moduleOrError =
333 llvm::parseBitcodeFile(buffer->getMemBufferRef(), llvmContext);
334 if (moduleOrError.takeError())
335 throw std::runtime_error("Could not parse bitcode file");
336 auto module = std::move(moduleOrError.get());
337 for (llvm::Function &func : *module) {
338 if (func.hasFnAttribute("entry_point") &&
339 func.hasFnAttribute("output_names")) {
340 output_names = nlohmann::json::parse(
341 func.getFnAttribute("output_names").getValueAsString());
342 break;
343 }
344 }
345 }
346 }
347 return output_names;
348 }

354 std::vector<cudaq::KernelExecution>
355 lowerQuakeCode(const std::string &kernelName, void *kernelArgs) {
356
357 auto [m_module, contextPtr, updatedArgs] =
358 extractQuakeCodeAndContext(kernelName, kernelArgs);
359
360 mlir::MLIRContext &context = *contextPtr;
361
362 // Extract the kernel name
363 auto func = m_module.lookupSymbol<mlir::func::FuncOp>(
364 std::string("__nvqpp__mlirgen__") + kernelName);
365
366 // Create a new Module to clone the function into
367 auto location = mlir::FileLineColLoc::get(&context, "<builder>", 1, 1);
368 mlir::ImplicitLocOpBuilder builder(location, &context);
369
370 // FIXME this should be added to the builder.
371 if (!func->hasAttr(cudaq::entryPointAttrName))
372 func->setAttr(cudaq::entryPointAttrName, builder.getUnitAttr());
373 auto moduleOp = builder.create<mlir::ModuleOp>();
374 moduleOp.push_back(func.clone());
375 moduleOp->setAttrs(m_module->getAttrDictionary());
376
377 for (auto &op : m_module.getOps()) {
378 // Add any global symbols, including global constant arrays.
379 // Global constant arrays can be created during compilation,
380 // `lift-array-value`, `quake-synthesizer`, and `get-concrete-matrix`
381 // passes.
382 if (auto globalOp = dyn_cast<cudaq::cc::GlobalOp>(op))
383 moduleOp.push_back(globalOp.clone());
384 }
385
386 // Lambda to apply a specific pipeline to the given ModuleOp
387 auto runPassPipeline = [&](const std::string &pipeline,
388 mlir::ModuleOp moduleOpIn) {
389 mlir::PassManager pm(&context);
390 std::string errMsg;
391 llvm::raw_string_ostream os(errMsg);
392 cudaq::info("Pass pipeline for {} = {}", kernelName, pipeline);
393 if (failed(parsePassPipeline(pipeline, pm, os)))
394 throw std::runtime_error(
395 "Remote rest platform failed to add passes to pipeline (" + errMsg +
396 ").");
397 if (disableMLIRthreading || enablePrintMLIREachPass)
398 moduleOpIn.getContext()->disableMultithreading();
399 if (enablePrintMLIREachPass)
400 pm.enableIRPrinting();
401 if (failed(pm.run(moduleOpIn)))
402 throw std::runtime_error("Remote rest platform Quake lowering failed.");
403 };
404
405 if (updatedArgs) {
406 cudaq::info("Run Quake Synth.\n");
407 mlir::PassManager pm(&context);
408 pm.addPass(cudaq::opt::createQuakeSynthesizer(kernelName, updatedArgs));
409 pm.addPass(mlir::createCanonicalizerPass());
410 if (disableMLIRthreading || enablePrintMLIREachPass)
411 moduleOp.getContext()->disableMultithreading();
412 if (enablePrintMLIREachPass)
413 pm.enableIRPrinting();
414 if (failed(pm.run(moduleOp)))
415 throw std::runtime_error("Could not successfully apply quake-synth.");
416 }
417
418 runPassPipeline(passPipelineConfig, moduleOp);
419
420 auto entryPointFunc = moduleOp.lookupSymbol<mlir::func::FuncOp>(
421 std::string("__nvqpp__mlirgen__") + kernelName);
422 std::vector<std::size_t> mapping_reorder_idx;
423 if (auto mappingAttr = dyn_cast_if_present<mlir::ArrayAttr>(
424 entryPointFunc->getAttr("mapping_reorder_idx"))) {
425 mapping_reorder_idx.resize(mappingAttr.size());
426 std::transform(mappingAttr.begin(), mappingAttr.end(),
427 mapping_reorder_idx.begin(), [](mlir::Attribute attr) {
428 return mlir::cast<mlir::IntegerAttr>(attr).getInt();
429 });
430 }
431
432 if (executionContext) {
433 if (executionContext->name == "sample")
434 executionContext->reorderIdx = mapping_reorder_idx;
435 else
436 executionContext->reorderIdx.clear();
437 }
438
439 std::vector<std::pair<std::string, mlir::ModuleOp>> modules;
440 // Apply observations if necessary
441 if (executionContext && executionContext->name == "observe") {
442 mapping_reorder_idx.clear();
443 runPassPipeline("canonicalize,cse", moduleOp);
444 cudaq::spin_op &spin = *executionContext->spin.value();
445 for (const auto &term : spin) {
446 if (term.is_identity())
447 continue;
448
449 // Get the ansatz
450 auto ansatz = moduleOp.lookupSymbol<mlir::func::FuncOp>(
451 std::string("__nvqpp__mlirgen__") + kernelName);
452
453 // Create a new Module to clone the ansatz into it
454 auto tmpModuleOp = builder.create<mlir::ModuleOp>();
455 tmpModuleOp.push_back(ansatz.clone());
456
457 // Extract the binary symplectic encoding
458 auto [binarySymplecticForm, coeffs] = term.get_raw_data();
459
460 // Create the pass manager, add the quake observe ansatz pass
461 // and run it followed by the canonicalizer
462 mlir::PassManager pm(&context);
463 pm.addNestedPass<mlir::func::FuncOp>(
464 cudaq::opt::createObserveAnsatzPass(binarySymplecticForm[0]));
465 if (disableMLIRthreading || enablePrintMLIREachPass)
466 tmpModuleOp.getContext()->disableMultithreading();
467 if (enablePrintMLIREachPass)
468 pm.enableIRPrinting();
469 if (failed(pm.run(tmpModuleOp)))
470 throw std::runtime_error("Could not apply measurements to ansatz.");
471 runPassPipeline(passPipelineConfig, tmpModuleOp);
472 modules.emplace_back(term.to_string(false), tmpModuleOp);
473 }
474 } else
475 modules.emplace_back(kernelName, moduleOp);
476
477 if (emulate) {
478 // If we are in emulation mode, we need to first get a
479 // full QIR representation of the code. Then we'll map to
480 // an LLVM Module, create a JIT ExecutionEngine pointer
481 // and use that for execution
482 for (auto &[name, module] : modules) {
483 auto clonedModule = module.clone();
484 jitEngines.emplace_back(
485 cudaq::createQIRJITEngine(clonedModule, codegenTranslation));
486 }
487 }
488
489 // Get the code gen translation
490 auto translation = cudaq::getTranslation(codegenTranslation);
491
492 // Apply user-specified codegen
493 std::vector<cudaq::KernelExecution> codes;
494 for (auto &[name, moduleOpI] : modules) {
495 std::string codeStr;
496 {
497 llvm::raw_string_ostream outStr(codeStr);
498 if (disableMLIRthreading)
499 moduleOpI.getContext()->disableMultithreading();
500 if (failed(translation(moduleOpI, outStr, postCodeGenPasses, printIR,
501 enablePrintMLIREachPass)))
502 throw std::runtime_error("Could not successfully translate to " +
503 codegenTranslation + ".");
504 }
505
506 // Form an output_names mapping from codeStr
507 nlohmann::json j = formOutputNames(codegenTranslation, codeStr);
508
509 codes.emplace_back(name, codeStr, j, mapping_reorder_idx);
510 }
511
512 cleanupContext(contextPtr);
513 return codes;
514 }

520 void launchKernel(const std::string &kernelName, void (*kernelFunc)(void *),
521 void *args, std::uint64_t voidStarSize,
522 std::uint64_t resultOffset) override {
523 cudaq::info("launching remote rest kernel ({})", kernelName);
524
525 // TODO future iterations of this should support non-void return types.
526 if (!executionContext)
527 throw std::runtime_error(
528 "Remote rest execution can only be performed via cudaq::sample(), "
529 "cudaq::observe(), or cudaq::draw().");
530
531 // Get the Quake code, lowered according to config file.
532 auto codes = lowerQuakeCode(kernelName, args);
533
534 // After performing lowerQuakeCode, check to see if we are simply drawing
535 // the circuit. If so, perform the trace here and then return.
536 if (executionContext->name == "tracer" && jitEngines.size() == 1) {
537 cudaq::getExecutionManager()->setExecutionContext(executionContext);
538 invokeJITKernelAndRelease(jitEngines[0], kernelName);
539 cudaq::getExecutionManager()->resetExecutionContext();
540 jitEngines.clear();
541 return;
542 }
543
544 // Get the current execution context and number of shots
545 std::size_t localShots = 1000;
546 if (executionContext->shots != std::numeric_limits<std::size_t>::max() &&
547 executionContext->shots != 0)
548 localShots = executionContext->shots;
549
550 executor->setShots(localShots);
551
552 // If emulation requested, then just grab the function
553 // and invoke it with the simulator
554 cudaq::details::future future;
555 if (emulate) {
556
557 // Fetch the thread-specific seed outside and then pass it inside.
558 std::size_t seed = cudaq::get_random_seed();
559
560 // Launch the execution of the simulated jobs asynchronously
561 future = cudaq::details::future(std::async(
562 std::launch::async,
563 [&, codes, localShots, kernelName, seed,
564 reorderIdx = executionContext->reorderIdx,
565 localJIT = std::move(jitEngines)]() mutable -> cudaq::sample_result {
566 std::vector<cudaq::ExecutionResult> results;
567
568 // If seed is 0, then it has not been set.
569 if (seed > 0)
570 cudaq::set_random_seed(seed);
571
572 bool hasConditionals =
573 cudaq::kernelHasConditionalFeedback(kernelName);
574 if (hasConditionals && codes.size() > 1)
575 throw std::runtime_error("error: spin_ops not yet supported with "
576 "kernels containing conditionals");
577 if (hasConditionals) {
578 executor->setShots(1); // run one shot at a time
579
580 // If this is adaptive profile and the kernel has conditionals,
581 // then you have to run the code localShots times instead of
582 // running the kernel once and sampling the state localShots
583 // times.
584 if (hasConditionals) {
585 // Populate `counts` one shot at a time
586 cudaq::sample_result counts;
587 for (std::size_t shot = 0; shot < localShots; shot++) {
588 cudaq::ExecutionContext context("sample", 1);
589 context.hasConditionalsOnMeasureResults = true;
590 cudaq::getExecutionManager()->setExecutionContext(&context);
591 invokeJITKernel(localJIT[0], kernelName);
592 cudaq::getExecutionManager()->resetExecutionContext();
593 counts += context.result;
594 }
595 // Process `counts` and store into `results`
596 for (auto &regName : counts.register_names()) {
597 results.emplace_back(counts.to_map(regName), regName);
598 results.back().sequentialData =
599 counts.sequential_data(regName);
600 }
601 }
602 }
603
604 for (std::size_t i = 0; i < codes.size(); i++) {
605 cudaq::ExecutionContext context("sample", localShots);
606 context.reorderIdx = reorderIdx;
607 cudaq::getExecutionManager()->setExecutionContext(&context);
608 invokeJITKernelAndRelease(localJIT[i], kernelName);
609 cudaq::getExecutionManager()->resetExecutionContext();
610
611 // If there are multiple codes, this is likely a spin_op.
612 // If so, use the code name instead of the global register.
613 if (codes.size() > 1) {
614 results.emplace_back(context.result.to_map(), codes[i].name);
615 results.back().sequentialData =
616 context.result.sequential_data();
617 } else {
618 // For each register, add the context results into result.
619 for (auto &regName : context.result.register_names()) {
620 results.emplace_back(context.result.to_map(regName), regName);
621 results.back().sequentialData =
622 context.result.sequential_data(regName);
623 }
624 }
625 }
626 localJIT.clear();
627 return cudaq::sample_result(results);
628 }));
629
630 } else {
631 // Execute the codes produced in quake lowering
632 // Allow developer to disable remote sending (useful for debugging IR)
633 if (getEnvBool("DISABLE_REMOTE_SEND", false))
634 return;
635 else
636 future = executor->execute(codes);
637 }
638
639 // Keep this asynchronous if requested
640 if (executionContext->asyncExec) {
641 executionContext->futureResult = future;
642 return;
643 }
644
645 // Otherwise make this synchronous
646 executionContext->result = future.get();
647 }
