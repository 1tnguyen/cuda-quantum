67 template <typename KernelT, typename ArgsMapper>
68 gradient(KernelT &kernel, ArgsMapper &&argsMapper) {
69 ansatz_functor = [&](std::vector<double> x) {
70 auto as_args = argsMapper(x);
71 std::apply(
72 [&](auto &&...new_args) {
73 cudaq::invokeKernel(std::forward<KernelT>(kernel), new_args...);
74 },
75 as_args);
76 };

83 template <typename QuantumKernel, typename... Args>
84 void setArgs(QuantumKernel &kernel, Args &&...args) {
85 static_assert(
86 std::is_invocable_v<QuantumKernel, std::vector<double>, Args...>,
87 "Kernel must be invocable with std::vector<double> and Args...");
88 // Serialize all the parameters except for the first std::vector<double>
89 // parameter. The serialized ones will be saved and used later during each
90 // ansatz_functor invocation.
91 serializedArgs = serializeArgs(std::forward<Args>(args)...);
92 ansatz_functor = [&](std::vector<double> x) {
93 cudaq::invokeCallableWithSerializedArgs_vec<QuantumKernel,
94 std::decay_t<Args>...>(
95 x, serializedArgs.data(), serializedArgs.size(),
96 std::forward<QuantumKernel>(kernel));
97 };
98 }

112 template <typename KernelT, typename = std::enable_if_t<std::is_invocable_v<
113 KernelT, std::vector<double>>>>
114 gradient(KernelT &kernel) {
115 if constexpr (has_name<KernelT>::value)
116 if (kernel.getNumParams() != 1)
117 throw std::invalid_argument(
118 "Callable kernel from cudaq::make_kernel must "
119 "have 1 std::vector<double> argument. Provide an ArgMapper if "
120 "not.");
121 ansatz_functor = [&](std::vector<double> x) {
122 return cudaq::invokeKernel(std::forward<KernelT>(kernel), x);
123 };
