317struct ArgumentValidator {

319 T &val) {
320 // Default case, do nothing for now
321 argCounter++;
322 }
323};

328template <typename T>
329struct ArgumentValidator<std::vector<T>> {
330 static void validate(std::size_t &argCounter, std::vector<QuakeValue> &args,
331 std::vector<T> &input) {
332 if (argCounter >= args.size())
333 throw std::runtime_error("Error validating stdvec input to "
334 "kernel_builder. argCounter >= args.size()");
335
336 // Get the argument, increment the counter
337 auto &arg = args[argCounter];
338 argCounter++;
339
340 // Validate the input vector<T> if possible. If getRequiredElements()
341 // returns 0, any size vector is ok.
342 auto nRequiredElements = arg.getRequiredElements();
343 if (nRequiredElements && arg.canValidateNumElements() &&
344 input.size() < nRequiredElements)
345 throw std::runtime_error(
346 "Invalid vector<T> input. Number of elements provided (" +
347 std::to_string(input.size()) + ") != number of elements required (" +
348 std::to_string(nRequiredElements) + ").\n");
349 }
350};

723 void exp_pauli(const ParamT &theta, const QuakeValue &qubits,
