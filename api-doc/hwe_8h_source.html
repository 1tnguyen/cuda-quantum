24struct cnot_coupling {
25 std::size_t source;
26 std::size_t target;
27 cnot_coupling(std::size_t s, std::size_t t) : source(s), target(t) {}
28};

31inline std::vector<cnot_coupling> default_cnot_coupling(std::size_t numQubits) {
32 std::vector<cnot_coupling> cnotCoupling;
33 cnotCoupling.reserve(numQubits - 1);
34 for (std::size_t q = 0; q < numQubits - 1; q++)
35 cnotCoupling.push_back({q, q + 1});
36
37 return cnotCoupling;
38}

44__qpu__ void hwe(cudaq::qview<> qubits, std::size_t numLayers,
45 std::vector<double> parameters,
46 const std::vector<cnot_coupling> &cnotCoupling) {
47 std::size_t numQubits = qubits.size();
48
49 for (std::size_t thetaCounter = 0, i = 0; i < numQubits; i++) {
50 ry(parameters[thetaCounter], qubits[i]);
51 rz(parameters[thetaCounter + 1], qubits[i]);
52 thetaCounter += 2;
53 }
54
55 for (std::size_t thetaCounter = numQubits * 2, layer = 0; layer < numLayers;
56 layer++) {
57 for (auto &cnot : cnotCoupling)
58 x<cudaq::ctrl>(qubits[cnot.source], qubits[cnot.target]);
59
60 for (std::size_t q = 0; q < numQubits; q++) {
61 ry(parameters[thetaCounter], qubits[q]);
62 rz(parameters[thetaCounter + 1], qubits[q]);
63 thetaCounter += 2;
64 }
65 }
66}
