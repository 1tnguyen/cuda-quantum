107 void resetExecutionContext() override {

109 synchronize();

114 // Do any final post-processing before

149 void startAdjointRegion() override { adjointQueueStack.emplace_back(); }

151 void endAdjointRegion() override {
152 assert(!adjointQueueStack.empty() && "There must be at least one queue");
153
154 auto adjointQueue = std::move(adjointQueueStack.back());
155 adjointQueueStack.pop_back();
156
157 // Select the queue to which these instructions will be added.
158 InstructionQueue *queue = adjointQueueStack.empty()
159 ? &instructionQueue
160 : &(adjointQueueStack.back());
161
162 std::reverse(adjointQueue.begin(), adjointQueue.end());
163 for (auto &instruction : adjointQueue)
164 queue->push_back(instruction);
165 }

178 void apply(const std::string_view gateName, const std::vector<double> &params,
179 const std::vector<cudaq::QuditInfo> &controls,
180 const std::vector<cudaq::QuditInfo> &targets,
181 bool isAdjoint = false, spin_op op = spin_op()) override {
182
183 // Make a copy of the name that we can mutate if necessary
184 std::string mutable_name(gateName);
185
186 // Make a copy of the parameters that we can mutate
187 std::vector<double> mutable_params = params;
188
189 // Create an array of controls, we will
190 // prepend any extra controls if in a control region
191 std::vector<cudaq::QuditInfo> mutable_controls;
192 for (auto &e : extraControlIds)
193 mutable_controls.emplace_back(2, e);
194
195 for (auto &e : controls)
196 mutable_controls.push_back(e);
197
198 std::vector<cudaq::QuditInfo> mutable_targets;
199 for (auto &t : targets)
200 mutable_targets.push_back(t);
201
202 // We need to check if we need take the adjoint of the operation. To do this
203 // we use a logical XOR between `isAdjoint` and whether the size of
204 // `adjointQueueStack` is even. The size of `adjointQueueStack` corresponds
205 // to the number of nested `cudaq::adjoint` calls. If the size is even, then
206 // we need to change the operation when `isAdjoint` is true. If the size is
207 // odd, then we need to change the operation when `isAdjoint` is false.
208 // (Adjoint modifiers cancel each other, e.g, `adj adj r1` is `r1`.)
209 //
210 // The cases:
211 // * not-adjoint, even number of `cudaq::adjoint` => _no_ need to change op
212 // * not-adjoint, odd number of `cudaq::adjoint` => change op
213 // * adjoint, even number of `cudaq::adjoint` => change op
214 // * adjoint, odd number `cudaq::adjoint` => _no_ need to change op
215 //
216 bool evenAdjointStack = (adjointQueueStack.size() % 2) == 0;
217 if (isAdjoint != !evenAdjointStack) {
218 for (std::size_t i = 0; i < params.size(); i++)
219 mutable_params[i] = -1.0 * params[i];
220 if (gateName == "t")
221 mutable_name = "tdg";
222 else if (gateName == "s")
223 mutable_name = "sdg";
224 }
225
226 if (!adjointQueueStack.empty()) {
227 // Add to the adjoint instruction queue
228 adjointQueueStack.back().emplace_back(
229 mutable_name, mutable_params, mutable_controls, mutable_targets, op);
230 return;
231 }
232
233 // Add to the instruction queue
234 instructionQueue.emplace_back(std::move(mutable_name), mutable_params,
235 mutable_controls, mutable_targets, op);
236 }

238 void synchronize() override {
239 for (auto &instruction : instructionQueue) {
240 if (!isInTracerMode()) {
241 executeInstruction(instruction);
242 continue;
243 }
244
245 auto &&[name, params, controls, targets, op] = instruction;
246 executionContext->kernelTrace.appendInstruction(name, params, controls,
247 targets);
248 }
249 instructionQueue.clear();
250 }

252 int measure(const cudaq::QuditInfo &target,
253 const std::string registerName = "") override {
254 if (isInTracerMode())
255 return 0;
256
257 // We hit a measure, need to exec / clear instruction queue
258 synchronize();
259
260 // Instruction executed, run the measure call
261 return measureQudit(target, registerName);
262 }

264 cudaq::SpinMeasureResult measure(cudaq::spin_op &op) override {
265 synchronize();
266 measureSpinOp(op);
267 return std::make_pair(executionContext->expectationValue.value(),
268 executionContext->result);
269 }

277 }
