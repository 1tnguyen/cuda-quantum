71 template <typename SpinOpType>
72 double expectation(SpinOpType term) {
73 static_assert(std::is_same_v<spin_op, std::remove_reference_t<SpinOpType>>,
74 "Must provide a one term spin_op");
75 // Pauli == Pauli II..III
76 // e.g. someone might check for <Z>, which
77 // on more than 1 qubit can be <ZIII...III>
78 auto numQubits = spinOp.num_qubits();
79 auto termStr = term.to_string(false);
80 // Expand the string representation of the term to match the number of
81 // qubits of the overall spin_op this result represents by appending
82 // identity ops.
83 if (!data.has_expectation(termStr))
84 for (std::size_t i = termStr.size(); i < numQubits; i++)
85 termStr += "I";
86 return data.expectation(termStr);
87 }

91 "argument structure.")]] double
92 exp_val_z(SpinOpType term) {
93 static_assert(std::is_same_v<spin_op, std::remove_reference_t<SpinOpType>>,
94 "Must provide a one term spin_op");
95 // Pauli == Pauli II..III
96 // e.g. someone might check for <Z>, which
97 // on more than 1 qubit can be <ZIII...III>
98 auto numQubits = spinOp.num_qubits();
99 auto termStr = term.to_string(false);
100 if (!data.has_expectation(termStr) && termStr.size() == 1 && numQubits > 1)
101 for (std::size_t i = 1; i < numQubits; i++)
102 termStr += "I";

109 template <typename SpinOpType>
110 sample_result counts(SpinOpType term) {
111 static_assert(std::is_same_v<spin_op, std::remove_reference_t<SpinOpType>>,
112 "Must provide a one term spin_op");
113 assert(term.num_terms() == 1 && "Must provide a one term spin_op");
114 auto numQubits = spinOp.num_qubits();
115 auto termStr = term.to_string(false);
116 if (!data.has_expectation(termStr) && termStr.size() == 1 && numQubits > 1)
117 for (std::size_t i = 1; i < numQubits; i++)
118 termStr += "I";
119 auto counts = data.to_map(termStr);
120 ExecutionResult result(counts);
121 return sample_result(result);
122 }
