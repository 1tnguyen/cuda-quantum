17// The state is represented by a quantum kernel.
18// For accessor APIs, we may resolve the state to a state vector by executing
19// the kernel on the remote simulator. For overlap API b/w 2 remote states, we
20// can send both kernels to the remote backend for execution and compute the
21// overlap.
22class RemoteSimulationState : public cudaq::SimulationState {
23protected:
24 std::string kernelName;
25 // Lazily-evaluated state data (just keeping the kernel name and arguments).
26 // e.g., to be evaluated at amplitude accessor APIs (const APIs, hence needs
27 // to be mutable) or overlap calculation with another remote state (combining
28 // the IR of both states for remote evaluation)
29 mutable std::unique_ptr<cudaq::SimulationState> state;
30 mutable std::vector<char> argsBuffer;
31 // Cache log messages from the remote execution.
32 // Mutable to support lazy execution during `const` API calls.
33 mutable std::string platformExecutionLog;
34
35public:
37 template <typename QuantumKernel, typename... Args>
38 RemoteSimulationState(QuantumKernel &&kernel, Args &&...args) {
39 if constexpr (has_name<QuantumKernel>::value) {
40 // kernel_builder kernel: need to JIT code to get it registered.
41 static_cast<cudaq::details::kernel_builder_base &>(kernel).jitCode();
42 kernelName = kernel.name();
43 } else {
44 kernelName = cudaq::getKernelName(kernel);
45 }
46
47 argsBuffer = cudaq::serializeArgs(std::forward<Args>(args)...);
48 }
