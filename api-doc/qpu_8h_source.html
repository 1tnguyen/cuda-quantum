58 void handleObservation(ExecutionContext *localContext) {
59 // The reason for the 2 if checks is simply to do a flushGateQueue() before
60 // initiating the trace.
61 bool execute = localContext && localContext->name == "observe";
62 if (execute) {
63 ScopedTraceWithContext(cudaq::TIMING_OBSERVE,
64 "handleObservation flushGateQueue()");
65 getExecutionManager()->flushGateQueue();
66 }
67 if (execute) {
68 ScopedTraceWithContext(cudaq::TIMING_OBSERVE,
69 "QPU::handleObservation (after flush)");
70 double sum = 0.0;
71 if (!localContext->spin.has_value())
72 throw std::runtime_error("[QPU] Observe ExecutionContext specified "
73 "without a cudaq::spin_op.");
74
75 std::vector<cudaq::ExecutionResult> results;
76 cudaq::spin_op &H = *localContext->spin.value();
77
78 // If the backend supports the observe task,
79 // let it compute the expectation value instead of
80 // manually looping over terms, applying basis change ops,
81 // and computing <ZZ..ZZZ>
82 if (localContext->canHandleObserve) {
83 auto [exp, data] = cudaq::measure(H);
84 localContext->expectationValue = exp;
85 localContext->result = data;
86 } else {
87
88 // Loop over each term and compute coeff * <term>
89 H.for_each_term([&](cudaq::spin_op &term) {
90 if (term.is_identity())
91 sum += term.get_coefficient().real();
92 else {
93 // This takes a longer time for the first iteration unless
94 // flushGateQueue() is called above.
95 auto [exp, data] = cudaq::measure(term);
96 results.emplace_back(data.to_map(), term.to_string(false), exp);
97 sum += term.get_coefficient().real() * exp;
98 }
99 });
100
101 localContext->expectationValue = sum;
102 localContext->result = cudaq::sample_result(sum, results);
103 }
104 }
105 }
