45bool setupTargetTriple(llvm::Module *llvmModule) {
46 // Setup the machine properties from the current architecture.
47 auto targetTriple = llvm::sys::getDefaultTargetTriple();
48 std::string errorMessage;
49 const auto *target =
50 llvm::TargetRegistry::lookupTarget(targetTriple, errorMessage);
51 if (!target)
52 return false;
53
54 std::string cpu(llvm::sys::getHostCPUName());
55 llvm::SubtargetFeatures features;
56 llvm::StringMap<bool> hostFeatures;
57
58 if (llvm::sys::getHostCPUFeatures(hostFeatures))
59 for (auto &f : hostFeatures)
60 features.AddFeature(f.first(), f.second);
61
62 std::unique_ptr<llvm::TargetMachine> machine(target->createTargetMachine(
63 targetTriple, cpu, features.getString(), {}, {}));
64 if (!machine)
65 return false;
66
67 llvmModule->setDataLayout(machine->createDataLayout());
68 llvmModule->setTargetTriple(targetTriple);
69
70 return true;
71}

73void optimizeLLVM(llvm::Module *module) {
74 auto optPipeline = mlir::makeOptimizingTransformer(
75 /*optLevel=*/3, /*sizeLevel=*/0,
76 /*targetMachine=*/nullptr);
77 if (auto err = optPipeline(module))
78 throw std::runtime_error("Failed to optimize LLVM IR ");
79
80 // Remove memory attributes from entry_point functions because the optimizer
81 // sometimes applies it to degenerate cases (empty programs), and IonQ cannot
82 // support that.
83 for (llvm::Function &func : *module)
84 if (func.hasFnAttribute("entry_point"))
85 func.removeFnAttr(llvm::Attribute::Memory);
86}

88void applyWriteOnlyAttributes(llvm::Module *llvmModule) {
89 // Note that we only need to inspect QIRMeasureBody because MeasureCallConv
90 // and MeasureToRegisterCallConv have already been called, so only
91 // QIRMeasureBody remains.
92 const unsigned int arg_num = 1;
93
94 // Apply attribute to measurement function declaration
95 if (auto func = llvmModule->getFunction(cudaq::opt::QIRMeasureBody)) {
96 func->addParamAttr(arg_num, llvm::Attribute::WriteOnly);
97 }
98
99 // Apply to measurement function calls
100 for (llvm::Function &func : *llvmModule)
101 for (llvm::BasicBlock &block : func)
102 for (llvm::Instruction &inst : block) {
103 auto callInst = llvm::dyn_cast_or_null<llvm::CallBase>(&inst);
104 if (callInst && callInst->getCalledFunction()) {
105 auto calledFunc = callInst->getCalledFunction();
106 auto funcName = calledFunc->getName();
107 if (funcName == cudaq::opt::QIRMeasureBody)
108 callInst->addParamAttr(arg_num, llvm::Attribute::WriteOnly);
109 }
110 }
111}

113// Once a call to a function with irreversible attribute is seen, no more calls
114// to reversible functions are allowed. This is somewhat of an implied
115// specification because the specification describes the program in terms of 4
116// sequential blocks. The 2nd block contains reversible operations, and the 3rd
117// block contains irreversible operations (measurements), and the blocks may not
118// overlap.
119// Reference:
120// https://github.com/qir-alliance/qir-spec/blob/main/specification/under_development/profiles/Base_Profile.md?plain=1#L237
121mlir::LogicalResult
122verifyBaseProfileMeasurementOrdering(llvm::Module *llvmModule) {
123 bool irreversibleSeenYet = false;
124 for (llvm::Function &func : *llvmModule)
125 for (llvm::BasicBlock &block : func)
126 for (llvm::Instruction &inst : block) {
127 auto callInst = llvm::dyn_cast_or_null<llvm::CallBase>(&inst);
128
129 if (callInst && callInst->getCalledFunction()) {
130 auto calledFunc = callInst->getCalledFunction();
131 auto funcName = calledFunc->getName();
132 bool isIrreversible = calledFunc->hasFnAttribute("irreversible");
133 bool isReversible = !isIrreversible;
134 bool isOutputFunction = funcName == cudaq::opt::QIRRecordOutput;
135 if (isReversible && !isOutputFunction && irreversibleSeenYet) {
136 llvm::errs() << "error: reversible function " << funcName
137 << " came after irreversible function\n";
138 return mlir::failure();
139 }
140 if (isIrreversible)
141 irreversibleSeenYet = true;
142 }
143 }
144 return mlir::success();
145}

147// Verify that output recording calls
148// 1) Have the nonnull attribute on any i8* parameters
149// 2) Have unique names
150mlir::LogicalResult verifyOutputCalls(llvm::CallBase *callInst,
151 std::set<std::string> &outputList) {
152 int iArg = 0;
153 for (auto &arg : callInst->args()) {
154 auto myArg = arg->getType();
155 auto ptrTy = dyn_cast_or_null<llvm::PointerType>(myArg);
156 // If we're dealing with the i8* parameters
157 if (ptrTy != nullptr &&
158 ptrTy->getNonOpaquePointerElementType()->isIntegerTy(8)) {
159 // Verify that it has the nonnull attribute
160 if (!callInst->paramHasAttr(iArg, llvm::Attribute::NonNull)) {
161 llvm::errs() << "error - nonnull attribute is missing from i8* "
162 "parameter of "
163 << cudaq::opt::QIRRecordOutput << " function\n";
164 return mlir::failure();
165 }
166
167 // Lookup the string value from IR that looks like this:
168 // clang-format off
169 // i8* nonnull getelementptr inbounds ([3 x i8], [3 x i8]* @cstr.723000, i64 0, i64 0))
170 // clang-format on
171 auto constExpr = llvm::dyn_cast_or_null<llvm::ConstantExpr>(arg);
172 if (constExpr &&
173 constExpr->getOpcode() == llvm::Instruction::GetElementPtr) {
174 llvm::Value *globalValue = constExpr->getOperand(0);
175 auto globalVar =
176 llvm::dyn_cast_or_null<llvm::GlobalVariable>(globalValue);
177
178 // Get the string value of the output name and compare it against
179 // the previously identified names in outputList[].
180 if (globalVar && globalVar->hasInitializer()) {
181 auto constDataArray = llvm::dyn_cast_or_null<llvm::ConstantDataArray>(
182 globalVar->getInitializer());
183 if (constDataArray) {
184 std::string strValue = constDataArray->getAsCString().str();
185 if (outputList.find(strValue) != outputList.end()) {
186 llvm::errs() << "error - duplicate output name (" << strValue
187 << ") found!\n";
188 return mlir::failure();
189 }
190 }
191 }
192 }
193 }
194
195 iArg++;
196 }
197 return mlir::success();
198}

200// Loop through the arguments in a call and verify that they are all constants
201mlir::LogicalResult verifyConstArguments(llvm::CallBase *callInst) {
202 int iArg = 0;
203 auto func = callInst ? callInst->getCalledFunction() : nullptr;
204 auto funcName = func ? func->getName() : "N/A";
205 for (auto &arg : callInst->args()) {
206 // Try casting to Constant Type. Fail if it's not a constant.
207 if (!dyn_cast_or_null<llvm::Constant>(arg)) {
208 llvm::errs() << "error: argument #" << iArg << " ('" << *arg
209 << "') in call " << funcName << " is not a constant\n";
210 return mlir::failure();
211 }
212 iArg++;
213 }
214 return mlir::success();
215}

217// Loop over the recording output functions and verify their characteristics
218mlir::LogicalResult verifyOutputRecordingFunctions(llvm::Module *llvmModule) {
219 for (llvm::Function &func : *llvmModule) {
220 std::set<std::string> outputList;
221 for (llvm::BasicBlock &block : func)
222 for (llvm::Instruction &inst : block) {
223 auto callInst = llvm::dyn_cast_or_null<llvm::CallBase>(&inst);
224 auto func = callInst ? callInst->getCalledFunction() : nullptr;
225 // All call arguments must be constants
226 if (func && failed(verifyConstArguments(callInst)))
227 return mlir::failure();
228 // If it's an output function, do additional verification
229 if (func && func->getName() == cudaq::opt::QIRRecordOutput)
230 if (failed(verifyOutputCalls(callInst, outputList)))
231 return mlir::failure();
232 }
233 }
234 return mlir::success();
235}

237// Convert a `nullptr` or `inttoptr (i64 1 to Ptr)` into an integer
238std::size_t getArgAsInteger(llvm::Value *arg) {
239 std::size_t ret = 0; // handles the nullptr case
240 // Now handle the `inttoptr (i64 1 to Ptr)` case
241 auto constValue = dyn_cast<llvm::Constant>(arg);
242 if (auto constExpr = dyn_cast<llvm::ConstantExpr>(constValue))
243 if (constExpr->getOpcode() == llvm::Instruction::IntToPtr)
244 if (auto constInt = dyn_cast<llvm::ConstantInt>(constExpr->getOperand(0)))
245 ret = constInt->getZExtValue();
246 return ret;
247}

259// Perform range checking on qubit and result values. This currently only checks
260// QIRMeasureBody and QIRRecordOutput. Checking more than that would
261// require comprehending the full list of possible QIS instructions, which is
262// not currently feasible.
263mlir::LogicalResult verifyQubitAndResultRanges(llvm::Module *llvmModule) {
264 std::size_t required_num_qubits = 0;
265 std::size_t required_num_results = 0;
266 for (llvm::Function &func : *llvmModule) {
267 if (func.hasFnAttribute("entry_point")) {
268 required_num_qubits = func.getFnAttributeAsParsedInteger(
269 "requiredQubits", required_num_qubits);
270 required_num_results = func.getFnAttributeAsParsedInteger(
271 "requiredResults", required_num_results);
272 break; // no need to keep looking
273 }
274 }
275 for (llvm::Function &func : *llvmModule) {
276 for (llvm::BasicBlock &block : func) {
277 for (llvm::Instruction &inst : block) {
278 if (auto callInst = llvm::dyn_cast_or_null<llvm::CallBase>(&inst)) {
279 if (auto func = callInst->getCalledFunction()) {
280 // All results must be in range for output recording functions
281 if (func->getName() == cudaq::opt::QIRRecordOutput) {
282 auto result = getArgAsInteger(callInst->getArgOperand(0));
283 CHECK_RANGE(result, required_num_results);
284 }
285 // All qubits and results must be in range for measurements
286 else if (func->getName() == cudaq::opt::QIRMeasureBody) {
287 auto qubit = getArgAsInteger(callInst->getArgOperand(0));
288 auto result = getArgAsInteger(callInst->getArgOperand(1));
289 CHECK_RANGE(qubit, required_num_qubits);
290 CHECK_RANGE(result, required_num_results);
291 }
292 }
293 }
294 }
295 }
296 }
297 return mlir::success();
298}

300// Verify that only the allowed LLVM instructions are present
301mlir::LogicalResult verifyLLVMInstructions(llvm::Module *llvmModule,
302 bool isBaseProfile) {
303 bool isAdaptiveProfile = !isBaseProfile;
304 for (llvm::Function &func : *llvmModule)
305 for (llvm::BasicBlock &block : func)
306 for (llvm::Instruction &inst : block) {
307 // Only specific instructions are allowed at the top level, depending on
308 // the specific profile
309 bool isValidBaseProfileInstruction =
310 llvm::isa<llvm::CallBase>(inst) ||
311 llvm::isa<llvm::BranchInst>(inst) ||
312 llvm::isa<llvm::ReturnInst>(inst);
313 // Note: there is an outstanding question about the adaptive profile
314 // with respect to `switch` and `select` instructions. They are
315 // currently described as "optional" in the spec, but there is no way to
316 // specify their presence via module flags. So to be cautious, for now
317 // we will assume they are not allowed in cuda-quantum programs.
318 bool isValidAdaptiveProfileInstruction = isValidBaseProfileInstruction;
319 // bool isValidAdaptiveProfileInstruction =
320 // isValidBaseProfileInstruction ||
321 // llvm::isa<llvm::SwitchInst>(inst) ||
322 // llvm::isa<llvm::SelectInst>(inst);
323 if (isBaseProfile && !isValidBaseProfileInstruction) {
324 llvm::errs() << "error - invalid instruction found: " << inst << '\n';
325 return mlir::failure();
326 } else if (isAdaptiveProfile && !isValidAdaptiveProfileInstruction) {
327 llvm::errs() << "error - invalid instruction found: " << inst << '\n';
328 return mlir::failure();
329 }
330 // Only inttoptr and getelementptr instructions are present as inlined
331 // call argument operations. These instructions may not be present
332 // unless they inlined call argument operations.
333 auto call = llvm::dyn_cast_or_null<llvm::CallBase>(&inst);
334 if (call)
335 for (auto &arg : call->args()) {
336 auto constExpr = llvm::dyn_cast_or_null<llvm::ConstantExpr>(arg);
337 if (constExpr &&
338 constExpr->getOpcode() != llvm::Instruction::GetElementPtr &&
339 constExpr->getOpcode() != llvm::Instruction::IntToPtr) {
340 llvm::errs() << "error - invalid instruction found: "
341 << *constExpr << '\n';
342 return mlir::failure();
343 }
344 }
345 }
346 return mlir::success();
347}

355mlir::LogicalResult
356qirProfileTranslationFunction(const char *qirProfile, mlir::Operation *op,
357 llvm::raw_string_ostream &output,
358 const std::string &additionalPasses, bool printIR,
359 bool printIntermediateMLIR) {
360 ScopedTraceWithContext(cudaq::TIMING_JIT, "qirProfileTranslationFunction");
361
362 const std::uint32_t qir_major_version = 1;
363 const std::uint32_t qir_minor_version = 0;
364
365 const bool isAdaptiveProfile = std::string{qirProfile} == "qir-adaptive";
366 const bool isBaseProfile = !isAdaptiveProfile;
367
368 auto context = op->getContext();
369 mlir::PassManager pm(context);
370 if (printIntermediateMLIR)
371 pm.enableIRPrinting();
372 std::string errMsg;
373 llvm::raw_string_ostream errOs(errMsg);
374 cudaq::opt::addPipelineConvertToQIR(pm, qirProfile);
375 // Add additional passes if necessary
376 if (!additionalPasses.empty() &&
377 failed(parsePassPipeline(additionalPasses, pm, errOs)))
378 return mlir::failure();
379 mlir::DefaultTimingManager tm;
380 tm.setEnabled(cudaq::isTimingTagEnabled(cudaq::TIMING_JIT_PASSES));
381 auto timingScope = tm.getRootScope(); // starts the timer
382 pm.enableTiming(timingScope); // do this right before pm.run
383 if (failed(pm.run(op)))
384 return mlir::failure();
385 timingScope.stop();
386
387 auto llvmContext = std::make_unique<llvm::LLVMContext>();
388 llvmContext->setOpaquePointers(false);
389 auto llvmModule = translateModuleToLLVMIR(op, *llvmContext);
390
391 // Apply required attributes for the Base Profile
392 applyWriteOnlyAttributes(llvmModule.get());
393
394 // Add required module flags for the Base Profile
395 llvmModule->addModuleFlag(llvm::Module::ModFlagBehavior::Error,
396 "qir_major_version", qir_major_version);
397 llvmModule->addModuleFlag(llvm::Module::ModFlagBehavior::Max,
398 "qir_minor_version", qir_minor_version);
399 auto falseValue =
400 llvm::ConstantInt::getFalse(llvm::Type::getInt1Ty(*llvmContext));
401 llvmModule->addModuleFlag(llvm::Module::ModFlagBehavior::Error,
402 "dynamic_qubit_management", falseValue);
403 llvmModule->addModuleFlag(llvm::Module::ModFlagBehavior::Error,
404 "dynamic_result_management", falseValue);
405 if (isAdaptiveProfile) {
406 auto trueValue =
407 llvm::ConstantInt::getTrue(llvm::Type::getInt1Ty(*llvmContext));
408 llvmModule->addModuleFlag(llvm::Module::ModFlagBehavior::Error,
409 "qubit_resetting", trueValue);
410 llvmModule->addModuleFlag(llvm::Module::ModFlagBehavior::Error,
411 "classical_ints", falseValue);
412 llvmModule->addModuleFlag(llvm::Module::ModFlagBehavior::Error,
413 "classical_floats", falseValue);
414 llvmModule->addModuleFlag(llvm::Module::ModFlagBehavior::Error,
415 "classical_fixed_points", falseValue);
416 llvmModule->addModuleFlag(llvm::Module::ModFlagBehavior::Error,
417 "user_functions", falseValue);
418 llvmModule->addModuleFlag(llvm::Module::ModFlagBehavior::Error,
419 "dynamic_float_args", falseValue);
420 llvmModule->addModuleFlag(llvm::Module::ModFlagBehavior::Error,
421 "extern_functions", falseValue);
422 llvmModule->addModuleFlag(llvm::Module::ModFlagBehavior::Error,
423 "backwards_branching", falseValue);
424 }
425
426 // Note: optimizeLLVM is the one that is setting nonnull attributes on
427 // the @__quantum__rt__result_record_output calls.
428 cudaq::optimizeLLVM(llvmModule.get());
429 if (!cudaq::setupTargetTriple(llvmModule.get()))
430 throw std::runtime_error("Failed to setup the llvm module target triple.");
431
432 // PyQIR currently requires named blocks. It's not clear if blocks can share
433 // names across functions, so we are being conservative by giving every block
434 // in the module a unique name for now.
435 int blockCounter = 0;
436 for (llvm::Function &func : *llvmModule)
437 for (llvm::BasicBlock &block : func)
438 if (!block.hasName())
439 block.setName(std::to_string(blockCounter++));
440
441 if (printIR)
442 llvm::errs() << *llvmModule;
443
444 if (failed(verifyOutputRecordingFunctions(llvmModule.get())))
445 return mlir::failure();
446
447 if (isBaseProfile &&
448 failed(verifyBaseProfileMeasurementOrdering(llvmModule.get())))
449 return mlir::failure();
450
451 if (failed(verifyQubitAndResultRanges(llvmModule.get())))
452 return mlir::failure();
453
454 if (failed(verifyLLVMInstructions(llvmModule.get(), isBaseProfile)))
455 return mlir::failure();
456
457 // Map the LLVM Module to Bitcode that can be submitted
458 llvm::SmallString<1024> bitCodeMem;
459 llvm::raw_svector_ostream os(bitCodeMem);
460 llvm::WriteBitcodeToFile(*llvmModule, os);
461 output << llvm::encodeBase64(bitCodeMem.str());
462 return mlir::success();
463}

542mlir::ExecutionEngine *createQIRJITEngine(mlir::ModuleOp &moduleOp,
543 llvm::StringRef convertTo) {
544 // The "fast" instruction selection compilation algorithm is actually very
545 // slow for large quantum circuits. Disable that here. Revisit this
546 // decision by testing large UCCSD circuits if jitCodeGenOptLevel is changed
547 // in the future. Also note that llvm::TargetMachine::setFastIsel() and
548 // setO0WantsFastISel() do not retain their values in our current version of
549 // LLVM. This use of LLVM command line parameters could be changed if the LLVM
550 // JIT ever supports the TargetMachine options in the future.
551 ScopedTraceWithContext(cudaq::TIMING_JIT, "createQIRJITEngine");
552 const char *argv[] = {"", "-fast-isel=0", nullptr};
553 llvm::cl::ParseCommandLineOptions(2, argv);
554
555 mlir::ExecutionEngineOptions opts;
556 opts.transformer = [](llvm::Module *m) { return llvm::ErrorSuccess(); };
557 opts.jitCodeGenOptLevel = llvm::CodeGenOpt::None;
558 opts.llvmModuleBuilder =
559 [convertTo = convertTo.str()](
560 mlir::Operation *module,
561 llvm::LLVMContext &llvmContext) -> std::unique_ptr<llvm::Module> {
562 ScopedTraceWithContext(cudaq::TIMING_JIT,
563 "createQIRJITEngine::llvmModuleBuilder");
564 llvmContext.setOpaquePointers(false);
565
566 auto *context = module->getContext();
567 mlir::PassManager pm(context);
568 std::string errMsg;
569 llvm::raw_string_ostream errOs(errMsg);
570 // Even though we're not lowering all the way to a real QIR profile for this
571 // emulated path, we need to pass in the `convertTo` in order to mimic what
572 // the non-emulated path would do.
573 cudaq::opt::commonPipelineConvertToQIR(pm, convertTo);
574 mlir::DefaultTimingManager tm;
575 tm.setEnabled(cudaq::isTimingTagEnabled(cudaq::TIMING_JIT_PASSES));
576 auto timingScope = tm.getRootScope(); // starts the timer
577 pm.enableTiming(timingScope); // do this right before pm.run
578 if (failed(pm.run(module)))
579 throw std::runtime_error(
580 "[createQIRJITEngine] Lowering to QIR for remote emulation failed.");
581 timingScope.stop();
582 auto llvmModule = translateModuleToLLVMIR(module, llvmContext);
583 if (!llvmModule)
584 throw std::runtime_error(
585 "[createQIRJITEngine] Lowering to LLVM IR failed.");
586
587 mlir::ExecutionEngine::setupTargetTriple(llvmModule.get());
588 return llvmModule;
589 };
590
591 auto jitOrError = mlir::ExecutionEngine::create(moduleOp, opts);
592 assert(!!jitOrError && "ExecutionEngine creation failed.");
593 return jitOrError.get().release();
594}
