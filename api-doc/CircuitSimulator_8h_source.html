29// @brief Collect summary data and print upon simulator termination
30struct SummaryData {
31 std::size_t gateCount = 0;
32 std::size_t controlCount = 0;
33 std::size_t targetCount = 0;
34 std::size_t svIO = 0;
35 std::size_t svFLOPs = 0;
36 bool enabled = false;
37 std::string name;
38 SummaryData() {
39 if (cudaq::isTimingTagEnabled(cudaq::TIMING_GATE_COUNT))
40 enabled = true;
41 }
42
44 void svGateUpdate(const std::size_t nControls, const std::size_t nTargets,
45 const std::size_t stateDimension,
46 const std::size_t stateVectorSizeBytes) {
47 assert(nControls <= 63);
48 if (enabled) {
49 gateCount++;
50 controlCount += nControls;
51 targetCount += nTargets;
52 // Times 2 because operating on the state vector requires both reading
53 // and writing.
54 svIO += (2 * stateVectorSizeBytes) / (1 << nControls);
55 // For each element of the state vector, 2 complex multiplies and 1
56 // complex accumulate is needed. This is reduced if there if this is a
57 // controlled operation.
58 // Each complex multiply is 6 real ops.
59 // So 2 complex multiplies and 1 complex addition is 2*6+2 = 14 ops.
60 svFLOPs += stateDimension * (14 * nTargets) / (1 << nControls);
61 }
62 }
63
64 ~SummaryData() {
65 if (enabled) {
66 cudaq::log("CircuitSimulator '{}' Total Program Metrics [tag={}]:", name,
67 cudaq::TIMING_GATE_COUNT);
68 cudaq::log("Gate Count = {}", gateCount);
69 cudaq::log("Control Count = {}", controlCount);
70 cudaq::log("Target Count = {}", targetCount);
71 cudaq::log("State Vector I/O (GB) = {:.6f}",
72 static_cast<double>(svIO) / 1e9);
73 cudaq::log("State Vector GFLOPs = {:.6f}",
74 static_cast<double>(svFLOPs) / 1e9);
75 }
76 }
77};

135 virtual void applyExpPauli(double theta,
136 const std::vector<std::size_t> &controls,
137 const std::vector<std::size_t> &qubitIds,
138 const cudaq::spin_op &op) {
139 if (op.is_identity()) {
140 if (controls.empty()) {
141 // exp(i*theta*Id) is noop if this is not a controlled gate.
142 return;
143 } else {
144 // Throw an error if this exp_pauli(i*theta*Id) becomes a non-trivial
145 // gate due to control qubits.
146 // FIXME: revisit this once
147 // https://github.com/NVIDIA/cuda-quantum/issues/483 is implemented.
148 throw std::logic_error("Applying controlled global phase via exp_pauli "
149 "of identity operator is not supported");
150 }
151 }
152 flushGateQueue();
153 cudaq::info(" [CircuitSimulator decomposing] exp_pauli({}, {})", theta,
154 op.to_string(false));
155 std::vector<std::size_t> qubitSupport;
156 std::vector<std::function<void(bool)>> basisChange;
157 op.for_each_pauli([&](cudaq::pauli type, std::size_t qubitIdx) {
158 auto qId = qubitIds[qubitIdx];
159 if (type != cudaq::pauli::I)
160 qubitSupport.push_back(qId);
161
162 if (type == cudaq::pauli::Y)
163 basisChange.emplace_back([this, qId](bool reverse) {
164 rx(!reverse ? M_PI_2 : -M_PI_2, qId);
165 });
166 else if (type == cudaq::pauli::X)
167 basisChange.emplace_back([this, qId](bool) { h(qId); });
168 });
169
170 if (!basisChange.empty())
171 for (auto &basis : basisChange)
172 basis(false);
173
174 std::vector<std::pair<std::size_t, std::size_t>> toReverse;
175 for (std::size_t i = 0; i < qubitSupport.size() - 1; i++) {
176 x({qubitSupport[i]}, qubitSupport[i + 1]);
177 toReverse.emplace_back(qubitSupport[i], qubitSupport[i + 1]);
178 }
179
180 // Since this is a compute-action-uncompute type circuit, we only need to
181 // apply control on this rz gate.
182 rz(-2.0 * theta, controls, qubitSupport.back());
183
184 std::reverse(toReverse.begin(), toReverse.end());
185 for (auto &[i, j] : toReverse)
186 x({i}, j);
187
188 if (!basisChange.empty()) {
189 std::reverse(basisChange.begin(), basisChange.end());
190 for (auto &basis : basisChange)
191 basis(true);
192 }
193 }

491 bool handleBasicSampling(const std::size_t qubitIdx,
492 const std::string &regName) {
493 if (executionContext && executionContext->name == "sample" &&
494 !executionContext->hasConditionalsOnMeasureResults) {
495 // Add the qubit to the sampling list
496 sampleQubits.push_back(qubitIdx);
497
498 auto processForRegName = [&](const std::string &regStr) {
499 // Insert the sample qubit into the register name map
500 auto iter = registerNameToMeasuredQubit.find(regStr);
501 if (iter == registerNameToMeasuredQubit.end())
502 registerNameToMeasuredQubit.emplace(
503 regStr, std::vector<std::size_t>{qubitIdx});
504 else if (std::find(iter->second.begin(), iter->second.end(),
505 qubitIdx) == iter->second.end())
506 iter->second.push_back(qubitIdx);
507 };
508
509 // Insert into global register and named register (if it exists)
510 processForRegName(cudaq::GlobalRegisterName);
511 if (!regName.empty())
512 processForRegName(regName);
513
514 return true;
515 }
516
517 return false;
518 }

524 void handleSamplingWithConditionals(const std::size_t qubitIdx,
525 const std::string bitResult,
526 const std::string &registerName) {
527 // We still care about what qubit we are measuring if in the
528 // sample-conditional context
529 if (executionContext && executionContext->name == "sample" &&
530 executionContext->hasConditionalsOnMeasureResults) {
531 std::string mutableRegisterName = registerName;
532
533 // If no registerName, we'll just sample normally
534 if (registerName.empty()) {
535 // Either this is library mode and we have register names attached
536 // to the execution context
537 if (midCircuitSampleResults.size() <
538 executionContext->registerNames.size()) {
539 mutableRegisterName =
540 executionContext->registerNames[midCircuitSampleResults.size()];
541 } else {
542 // or no register names, in which case we'll just treat it as
543 // a regular sampled qubit and drop out
544 sampleQubits.push_back(qubitIdx);
545 return;
546 }
547 }
548
549 cudaq::info("Handling Sampling With Conditionals: {}, {}, {}", qubitIdx,
550 bitResult, mutableRegisterName);
551 // See if we've observed this register before, if not
552 // start a vector of bit results, if we have, add the
553 // bit result to the existing vector
554 auto iter = midCircuitSampleResults.find(mutableRegisterName);
555 if (iter == midCircuitSampleResults.end())
556 midCircuitSampleResults.emplace(mutableRegisterName,
557 std::vector<std::string>{bitResult});
558 else
559 iter->second.push_back(bitResult);
560
561 // If this register is the same as last time, then we are
562 // writing to a bit vector register (auto var = mz(qreg))
563 if (lastMidCircuitRegisterName == mutableRegisterName)
564 vectorRegisters.push_back(mutableRegisterName);
565
566 // Store the last register name
567 lastMidCircuitRegisterName = mutableRegisterName;
568 }

573 std::string gateToString(const std::string_view gateName,
574 const std::vector<std::size_t> &controls,
575 const std::vector<ScalarType> &parameters,
576 const std::vector<std::size_t> &targets) {
577 std::string angleStr = "";
578 if (!parameters.empty()) {
579 angleStr = std::to_string(parameters[0]);
580 for (std::size_t i = 1; i < parameters.size(); i++)
581 angleStr += ", " + std::to_string(parameters[i]);
582 angleStr += ", ";
583 }
584
585 std::stringstream bits, ret;
586 if (!controls.empty()) {
587 bits << controls[0];
588 for (size_t i = 1; i < controls.size(); i++) {
589 bits << ", " << controls[i];
590 }
591 bits << ", " << targets[0];
592 for (size_t i = 1; i < targets.size(); i++) {
593 bits << ", " << targets[i];
594 }
595 ret << "(apply) ctrl-" << gateName << "(" << angleStr << bits.str()
596 << ")";
597 } else {
598 bits << targets[0];
599 for (size_t i = 1; i < targets.size(); i++) {
600 bits << ", " << targets[i];
601 }
602 ret << "(apply) " << gateName << "(" << angleStr << bits.str() << ")";
603 }
604 return ret.str();
605 }

644 void flushAnySamplingTasks(bool force = false) {
645 if (sampleQubits.empty())
646 return;
647
648 if (executionContext->hasConditionalsOnMeasureResults && !force)
649 return;
650
651 // Sort the qubit indices
652 std::sort(sampleQubits.begin(), sampleQubits.end());
653 auto last = std::unique(sampleQubits.begin(), sampleQubits.end());
654 sampleQubits.erase(last, sampleQubits.end());
655
656 cudaq::info("Sampling the current state, with measure qubits = {}",
657 sampleQubits);
658
659 // Ask the subtype to sample the current state
660 auto execResult =
661 sample(sampleQubits, executionContext->hasConditionalsOnMeasureResults
662 ? 1
663 : executionContext->shots);
664
665 if (registerNameToMeasuredQubit.empty()) {
666 executionContext->result.append(execResult);
667 } else {
668
669 for (auto &[regName, qubits] : registerNameToMeasuredQubit) {
670 // Measurements are sorted according to qubit allocation order
671 std::sort(qubits.begin(), qubits.end());
672 auto last = std::unique(qubits.begin(), qubits.end());
673 qubits.erase(last, qubits.end());
674
675 // Find the position of the qubits we have in the result bit string
676 // Create a map of qubit to bit string location
677 std::unordered_map<std::size_t, std::size_t> qubitLocMap;
678 for (std::size_t i = 0; i < qubits.size(); i++) {
679 auto iter =
680 std::find(sampleQubits.begin(), sampleQubits.end(), qubits[i]);
681 auto idx = std::distance(sampleQubits.begin(), iter);
682 qubitLocMap.insert({qubits[i], idx});
683 }
684
685 cudaq::ExecutionResult tmp(regName);
686 for (auto &[bits, count] : execResult.counts) {
687 std::string b = "";
688 for (auto &qb : qubits)
689 b += bits[qubitLocMap[qb]];
690 tmp.appendResult(b, count);
691 }
692
693 executionContext->result.append(tmp);
694 }
695 }
696
697 sampleQubits.clear();
698 registerNameToMeasuredQubit.clear();
699 }

746 void flushGateQueueImpl() override {
747 while (!gateQueue.empty()) {
748 auto &next = gateQueue.front();
749 if (isStateVectorSimulator() && summaryData.enabled)
750 summaryData.svGateUpdate(
751 next.controls.size(), next.targets.size(), stateDimension,
752 stateDimension * sizeof(std::complex<ScalarType>));
753 applyGate(next);
754 if (executionContext && executionContext->noiseModel) {
755 std::vector<std::size_t> noiseQubits{next.controls.begin(),
756 next.controls.end()};
757 noiseQubits.insert(noiseQubits.end(), next.targets.begin(),
758 next.targets.end());
759 applyNoiseChannel(next.operationName, noiseQubits);
760 }
761 gateQueue.pop();
762 }

829 std::size_t allocateQubit() override {
830 // Get a new qubit index
831 auto newIdx = tracker.getNextIndex();
832 if (isInBatchMode()) {
833 batchModeCurrentNumQubits++;
834 // In batch mode, we might already have an allocated state that
835 // has been set to |0..0>. We can reuse it as is, if the next qubit
836 // index is smaller than number of qubits of this allocated state.
837 if (newIdx < nQubitsAllocated)
838 return newIdx;
839 }
840
841 cudaq::info("Allocating new qubit with idx {} (nQ={}, dim={})", newIdx,
842 nQubitsAllocated, stateDimension);
843
844 // Increment the number of qubits and set
845 // the new state dimension
846 previousStateDimension = stateDimension;
847 nQubitsAllocated++;
848 stateDimension = calculateStateDim(nQubitsAllocated);
849
850 if (!isInTracerMode())
851 // Tell the subtype to grow the state representation
852 addQubitToState();
853
854 // May be that the state grows enough that we
855 // want to handle observation via sampling
856 if (executionContext)
857 executionContext->canHandleObserve = canHandleObserve();
858
859 // return the new qubit index
860 return newIdx;

1022 void resetExecutionContext() override {
1023 // If null, do nothing
1024 if (!executionContext)
1025 return;
1026
1027 // Get the ExecutionContext name
1028 auto execContextName = executionContext->name;
1029
1030 // If we are sampling...
1031 if (execContextName.find("sample") != std::string::npos) {
1032 // Sample the state over the specified number of shots
1033 if (sampleQubits.empty()) {
1034 if (isInBatchMode())
1035 sampleQubits.resize(batchModeCurrentNumQubits);
1036 else
1037 sampleQubits.resize(nQubitsAllocated);
1038 std::iota(sampleQubits.begin(), sampleQubits.end(), 0);
1039 }
1040
1041 // Flush the queue if there are any gates to apply
1042 flushGateQueue();
1043
1044 // Flush any queued up sampling tasks
1045 flushAnySamplingTasks(/*force this*/ true);
1046
1047 // Handle the processing for any mid circuit measurements
1048 for (auto &m : midCircuitSampleResults) {
1049 // Get the register name and the vector of bit results
1050 auto regName = m.first;
1051 auto bitResults = m.second;
1052 cudaq::ExecutionResult counts(regName);
1053
1054 if (std::find(vectorRegisters.begin(), vectorRegisters.end(),
1055 regName) != vectorRegisters.end()) {
1056 // this is a vector register
1057 std::string bitStr = "";
1058 for (std::size_t j = 0; j < bitResults.size(); j++)
1059 bitStr += bitResults[j];
1060
1061 counts.appendResult(bitStr, 1);
1062
1063 } else {
1064 // Not a vector, collate all bits into a 1 qubit counts dict
1065 for (std::size_t j = 0; j < bitResults.size(); j++) {
1066 counts.appendResult(bitResults[j], 1);
1067 }
1068 }
1069 executionContext->result.append(counts);
1070 }
1071
1072 // Reorder the global register (if necessary). This might be necessary if
1073 // the mapping pass had run and we want to undo the shuffle that occurred
1074 // during mapping.
1075 if (!executionContext->reorderIdx.empty()) {
1076 executionContext->result.reorder(executionContext->reorderIdx);
1077 executionContext->reorderIdx.clear();
1078 }
1079
1080 // Clear the sample bits for the next run
1081 sampleQubits.clear();
1082 midCircuitSampleResults.clear();
1083 lastMidCircuitRegisterName = "";
1084 currentCircuitName = "";
1085 }
1086
1087 // Set the state data if requested.
1088 if (executionContext->name == "extract-state") {
1089 flushGateQueue();
1090 executionContext->simulationState = getSimulationState();
1091 }
1092
1093 // Deallocate the deferred qubits, but do so
1094 // without explicit qubit reset.
1095 for (auto &deferred : deferredDeallocation)
1096 tracker.returnIndex(deferred);
1097
1098 bool shouldSetToZero = isInBatchMode() && !isLastBatch();
1099 executionContext = nullptr;
1100
1101 // Reset the state if we've deallocated all qubits.
1102 if (tracker.allDeallocated()) {
1103 if (shouldSetToZero) {
1104 cudaq::info("In batch mode currently, reset state to |0>");
1105 // Do not deallocate the state, but reset it to |0>
1106 setToZeroState();
1107 } else {
1108 cudaq::info("Deallocated all qubits, reseting state vector.");
1109 // all qubits deallocated,
1110 deallocateState();
1111 }
1112 }
1113
1114 batchModeCurrentNumQubits = 0;
1115 deferredDeallocation.clear();
1116 }

1132 void applyCustomOperation(const std::vector<std::complex<double>> &matrix,
1133 const std::vector<std::size_t> &controls,
1134 const std::vector<std::size_t> &targets,
1135 const std::string_view customName) override {
1136 flushAnySamplingTasks();
1137 auto numRows = std::sqrt(matrix.size());
1138 auto numQubits = std::log2(numRows);
1139 std::vector<std::complex<ScalarType>> actual;
1140 if (numQubits > 1 && getQubitOrdering() != QubitOrdering::msb) {
1141 // Convert the matrix to LSB qubit ordering
1142 auto convertOrdering = [](std::size_t numQubits, std::size_t idx) {
1143 std::size_t newIdx = 0;
1144 for (std::size_t i = 0; i < numQubits; ++i)
1145 if (idx & (1ULL << i))
1146 newIdx |= (1ULL << ((numQubits - 1) - i));
1147 return newIdx;
1148 };
1149 actual.resize(matrix.size());
1150 for (std::size_t i = 0; i < numRows; i++) {
1151 for (std::size_t j = 0; j < numRows; j++) {
1152 auto k = convertOrdering(numQubits, i);
1153 auto l = convertOrdering(numQubits, j);
1154 if (!std::is_same_v<double, ScalarType>) {
1155 actual[i * numRows + j] =
1156 static_cast<std::complex<ScalarType>>(matrix[k * numRows + l]);
1157 } else {
1158 auto element = matrix[k * numRows + l];
1159 actual[i * numRows + j] =
1160 std::complex<ScalarType>(element.real(), element.imag());
1161 }
1162 }
1163 }
1164 } else {
1165 std::transform(matrix.begin(), matrix.end(), std::back_inserter(actual),
1166 [](auto &&element) -> std::complex<ScalarType> {
1167 if (!std::is_same_v<double, ScalarType>) {
1168 return static_cast<std::complex<ScalarType>>(element);
1169 } else {
1170 return std::complex<ScalarType>(element.real(),
1171 element.imag());
1172 }
1173 });
1174 }
1175 if (cudaq::details::should_log(cudaq::details::LogLevel::info))
1176 cudaq::info(gateToString(customName.empty() ? "unknown op" : customName,
1177 controls, {}, targets) +
1178 " = {}",
1179 matrix);
1180 enqueueGate("custom", actual, controls, targets, {});
1181 }

1324 void measureSpinOp(const cudaq::spin_op &op) override {
1325 flushGateQueue();
1326
1327 if (executionContext->canHandleObserve) {
1328 auto result = observe(*executionContext->spin.value());
1329 executionContext->expectationValue = result.expectation();
1330 executionContext->result = result.raw_data();
1331 return;
1332 }
1333
1334 assert(op.num_terms() == 1 && "Number of terms is not 1.");
1335
1336 cudaq::info("Measure {}", op.to_string(false));
1337 std::vector<std::size_t> qubitsToMeasure;
1338 std::vector<std::function<void(bool)>> basisChange;
1339 op.for_each_pauli([&](cudaq::pauli type, std::size_t qubitIdx) {
1340 if (type != cudaq::pauli::I)
1341 qubitsToMeasure.push_back(qubitIdx);
1342
1343 if (type == cudaq::pauli::Y)
1344 basisChange.emplace_back([&, qubitIdx](bool reverse) {
1345 rx(!reverse ? M_PI_2 : -M_PI_2, qubitIdx);
1346 });
1347 else if (type == cudaq::pauli::X)
1348 basisChange.emplace_back([&, qubitIdx](bool) { h(qubitIdx); });
1349 });
1350
1351 // Change basis, flush the queue
1352 if (!basisChange.empty()) {
1353 for (auto &basis : basisChange)
1354 basis(false);
1355
1356 flushGateQueue();
1357 }
1358
1359 // Get whether this is shots-based
1360 int shots = 0;
1361 if (executionContext->shots > 0)
1362 shots = executionContext->shots;
1363
1364 // Sample and give the data to the context
1365 cudaq::ExecutionResult result = sample(qubitsToMeasure, shots);
1366 executionContext->expectationValue = result.expectationValue;
1367 executionContext->result = cudaq::sample_result(result);
1368
1369 // Restore the state.
1370 if (!basisChange.empty()) {
1371 std::reverse(basisChange.begin(), basisChange.end());
1372 for (auto &basis : basisChange)
1373 basis(true);
1374
1375 flushGateQueue();
1376 }
