90template <typename T>
91class async_result {
92protected:
94 details::future result;
95
97 spin_op *spinOp = nullptr;
98
99public:
100 async_result() = default;
101 async_result(spin_op *s) : spinOp(s) {}
102 async_result(details::future &&f, spin_op *op = nullptr)
103 : result(std::move(f)), spinOp(op) {}
104
107 T get() {
108 auto data = result.get();
109
110 if constexpr (std::is_same_v<T, sample_result>)
111 return data;
112
113 if constexpr (std::is_same_v<T, observe_result>) {
114 auto checkRegName = spinOp->to_string(false);
115 if (data.has_expectation(checkRegName))
116 return observe_result(data.expectation(checkRegName), *spinOp, data);
117
118 if (!spinOp)
119 throw std::runtime_error(
120 "Returning an observe_result requires a spin_op.");
121
122 // this assumes we ran in shots mode.
123 double sum = 0.0;
124 spinOp->for_each_term([&](spin_op &term) {
125 if (term.is_identity())
126 sum += term.get_coefficient().real();
127 else
128 sum += data.expectation(term.to_string(false)) *
129 term.get_coefficient().real();
130 });
131
132 return observe_result(sum, *spinOp, data);
133 }
134
135 return T();
136 }
137
138 template <typename U>
139 friend std::ostream &operator<<(std::ostream &, async_result<U> &);
140
141 template <typename U>
142 friend std::istream &operator>>(std::istream &, async_result<U> &);
143};
