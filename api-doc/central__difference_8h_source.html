15class central_difference : public gradient {
16public:
17 using gradient::gradient;
18 double step = 1e-4;
19
20 virtual std::unique_ptr<cudaq::gradient> clone() override {
21 auto newGrad = std::make_unique<central_difference>(*this);
22 newGrad->step = this->step;
23 return newGrad;
24 }
25
26 void compute(const std::vector<double> &x, std::vector<double> &dx,
27 const spin_op &h, double exp_h) override {
28 auto tmpX = x;
29 for (std::size_t i = 0; i < x.size(); i++) {
30 // increase value to x_i + dx_i
31 tmpX[i] += step;
32 auto px = getExpectedValue(tmpX, h);
33 // decrease the value to x_i - dx_i
34 tmpX[i] -= 2 * step;
35 auto mx = getExpectedValue(tmpX, h);
36 // return value back to x_i
37 tmpX[i] += step;
38 dx[i] = (px - mx) / (2. * step);
39 }
40 }

44 std::vector<double>
45 compute(const std::vector<double> &x,
46 const std::function<double(std::vector<double>)> &func,
47 double funcAtX) override {
48 std::vector<double> dx(x.size());
49 auto tmpX = x;
50 for (std::size_t i = 0; i < x.size(); i++) {
51 // increase value to x_i + dx_i
52 tmpX[i] += step;
53 double px = func(tmpX);
54 // decrease the value to x_i - dx_i
55 tmpX[i] -= 2 * step;
56 double mx = func(tmpX);
57 // return value back to x_i
58 tmpX[i] += step;
59 dx[i] = (px - mx) / (2. * step);
60 }

62 }
