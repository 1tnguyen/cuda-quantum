42 template <typename T, typename ScalarType = double>
43 auto getSizeAndPtrFromVec(const state_data &data) {
44 if constexpr (!std::is_same_v<T, ScalarType>)
45 throw std::runtime_error("[sim-state] invalid data precision.");
46
47 auto &vec = std::get<std::vector<std::complex<T>>>(data);
48 return std::make_tuple(
49 vec.size(),
50 reinterpret_cast<void *>(const_cast<std::complex<T> *>(vec.data())));
51 };
52
55 template <typename T, typename ScalarType = double>
56 auto getSizeAndPtrFromPair(const state_data &data) {
57 if constexpr (!std::is_same_v<T, ScalarType>)
58 throw std::runtime_error("[sim-state] invalid data precision.");
59
60 auto &pair = std::get<std::pair<std::complex<T> *, std::size_t>>(data);
61 return std::make_tuple(pair.second, reinterpret_cast<void *>(pair.first));
62 };
63
66 template <typename ScalarType = double>
67 auto getSizeAndPtr(const state_data &data) {
68 auto type = data.index();
69 std::tuple<std::size_t, void *> sizeAndPtr;
70 if (type == 0)
71 sizeAndPtr = getSizeAndPtrFromVec<double, ScalarType>(data);
72 else if (type == 1)
73 sizeAndPtr = getSizeAndPtrFromVec<float, ScalarType>(data);
74 else if (type == 2)
75 sizeAndPtr = getSizeAndPtrFromPair<double, ScalarType>(data);
76 else if (type == 3)
77 sizeAndPtr = getSizeAndPtrFromPair<float, ScalarType>(data);
78 else
79 throw std::runtime_error("unsupported data type for state.");
80
81 return sizeAndPtr;
82 }

97 struct Tensor {
98 void *data = nullptr;
99 std::vector<std::size_t> extents;
100 precision fp_precision;
101 std::size_t get_rank() const { return extents.size(); }
102 std::size_t get_num_elements() const {
103 std::size_t num = 1;
104 for (auto &el : extents)
105 num *= el;
106 return num;
107 }
108 std::size_t element_size() const {
109 return fp_precision == precision::fp32 ? sizeof(std::complex<float>)
110 : sizeof(std::complex<double>);

116 virtual std::unique_ptr<cudaq::SimulationState>
117 createFromData(const state_data &data) {
118 if (std::holds_alternative<TensorStateData>(data)) {
119 if (isArrayLike())
120 throw std::runtime_error(
121 "Cannot initialize state vector/density matrix state by matrix "
122 "product state tensors. Please use tensor network simulator "
123 "backends.");
124 auto &dataCasted = std::get<TensorStateData>(data);
125 return createFromSizeAndPtr(
126 dataCasted.size(),
127 const_cast<TensorStateData::value_type *>(dataCasted.data()),
128 data.index());
129 }
130 // Flat array state data
131 // Check the precision first. Get the size and
132 // data pointer from the input data.
133
134 if (getPrecision() == precision::fp32) {
135 auto [size, ptr] = getSizeAndPtr<float>(data);
136 return createFromSizeAndPtr(size, ptr, data.index());
137 }
138
139 auto [size, ptr] = getSizeAndPtr(data);
140 return createFromSizeAndPtr(size, ptr, data.index());
141 }
