34// The qspan represents a non-owning container of qudits. As such
35// it models both dynamically allocated qudit registers as well as
36// compile-time sized qudit registers.
37template <std::size_t N = dyn, std::size_t Levels = 2>
38class [[deprecated("The qspan type is deprecated in favor of qview.")]] qspan {
39public:
40 // Useful typedef exposing the underlying qudit type
41 // that this span contains.
42 using value_type = qudit<Levels>;
43
44#if CUDAQ_USE_STD20
45private:
46 // Reference to the non-owning span of qudits.
47 std::span<value_type, N> qudits;
48
49public:
50 // Construct a span that refers to the qudits in `other`.
51 template <typename R>
52 requires(std::ranges::range<R>)
53 qspan(R &&other) : qudits(other.begin(), other.end()) {}
54
55 // Copy constructor
56 qspan(qspan const &other) : qudits(other.qudits) {}
57
58 // Iterator interface.
59 auto begin() { return qudits.begin(); }
60 auto end() { return qudits.end(); }
61
62 // Returns the qudit at `idx`.
63 value_type &operator[](const std::size_t idx) { return qudits[idx]; }
64
65 // Returns the `[0, count)` qudits as a new span.
66 qspan<dyn, Levels> front(std::size_t count) { return qudits.first(count); }
67
68 // Returns the first qudit.
69 value_type &front() { return qudits.front(); }
70
71 // Returns the `[count, size())` qudits.
72 qspan<dyn, Levels> back(std::size_t count) { return qudits.last(count); }
73
74 // Returns the last qudit.
75 value_type &back() { return qudits.back(); }
76
77 // Returns the `[start, start+count)` qudits.
78 qspan<dyn, Levels> slice(std::size_t start, std::size_t count) {
79 return qudits.subspan(start, count);
80 }
81
82 // FIXME implement this
83 // Returns the `{start, start + stride, ...}` qudits.
84 // qspan<dyn, Levels>
85 // slice(std::size_t start, std::size_t stride, std::size_t end);
86
87 // Returns the number of contained qudits.
88 std::size_t size() const { return qudits.size(); }
89
90#else
91 // C++11 reimplementation of a std::span.
92
93private:
94 value_type *qudits = nullptr;
95 std::size_t qusize = 0;
96
97public:
98 qspan(value_type *otherQudits, std::size_t otherQusize)
99 : qudits(otherQudits), qusize(otherQusize) {}
100 template <typename Iterator>
101 qspan(Iterator &&otherQudits, std::size_t otherQusize)
102 : qudits(&*otherQudits), qusize(otherQusize) {}
103 template <typename R>
104 qspan(R &&other)
105 : qudits(&*other.begin()),
106 qusize(std::distance(other.begin(), other.end())) {}
107 qspan(qspan const &other) : qudits(other.qudits), qusize(other.qusize) {}
108
109 value_type *begin() { return qudits; }
110 value_type *end() { return qudits + qusize; }
111 value_type &operator[](const std::size_t idx) { return qudits[idx]; }
112 qspan<dyn, Levels> front(std::size_t count) { return {qudits, count}; }
113 value_type &front() { return *qudits; }
114 qspan<dyn, Levels> back(std::size_t count) {
115 return {qudits + qusize - count, count};
116 }
117 value_type &back() { return *(qudits + qusize - 1); }
118 qspan<dyn, Levels> slice(std::size_t start, std::size_t count) {
119 return {qudits + start, count};
120 }
121 std::size_t size() const { return qusize; }
122#endif
123};
124} // namespace cudaq
