25// Remote QPU: delegating the execution to a remotely-hosted server, which can
26// reinstate the execution context and JIT-invoke the kernel.
27class BaseRemoteSimulatorQPU : public cudaq::QPU {
28protected:
29 std::string m_simName;

32 std::unique_ptr<mlir::MLIRContext> m_mlirContext;
33 std::unique_ptr<cudaq::RemoteRuntimeClient> m_client;

37 cudaq::ExecutionContext *getExecutionContextForMyThread() {
38 std::scoped_lock<std::mutex> lock(m_contextMutex);
39 const auto iter = m_contexts.find(std::this_thread::get_id());
40 if (iter == m_contexts.end())
41 return nullptr;
42 return iter->second;
43 }

45public:
46 BaseRemoteSimulatorQPU()
47 : QPU(),
48 m_client(cudaq::registry::get<cudaq::RemoteRuntimeClient>("rest")) {}

51 virtual ~BaseRemoteSimulatorQPU() = default;

65 virtual void setTargetBackend(const std::string &backend) override {
66 auto parts = cudaq::split(backend, ';');
67 if (parts.size() % 2 != 0)
68 throw std::invalid_argument("Unexpected backend configuration string. "
69 "Expecting a ';'-separated key-value pairs.");
70 for (std::size_t i = 0; i < parts.size(); i += 2) {
71 if (parts[i] == "url")
72 m_client->setConfig({{"url", parts[i + 1]}});
73 if (parts[i] == "simulator")
74 m_simName = parts[i + 1];
75 }
76 }

83 void launchVQE(const std::string &name, const void *kernelArgs,
84 cudaq::gradient *gradient, cudaq::spin_op H,
85 cudaq::optimizer &optimizer, const int n_params,
86 const std::size_t shots) override {
87 cudaq::ExecutionContext *executionContextPtr =
88 getExecutionContextForMyThread();
89
90 if (executionContextPtr && executionContextPtr->name == "tracer")
91 return;
92
93 auto ctx = std::make_unique<ExecutionContext>("observe", shots);
94 ctx->kernelName = name;
95 ctx->spin = &H;
96 if (shots > 0)
97 ctx->shots = shots;
98
99 std::string errorMsg;
100 const bool requestOkay = m_client->sendRequest(
101 *m_mlirContext, *executionContextPtr, /*serializedCodeContext=*/nullptr,
102 gradient, &optimizer, n_params, m_simName, name, /*kernelFunc=*/nullptr,
103 kernelArgs, /*argSize=*/0, &errorMsg);
104 if (!requestOkay)
105 throw std::runtime_error("Failed to launch VQE. Error: " + errorMsg);
106 }

108 void launchKernel(const std::string &name, void (*kernelFunc)(void *),
109 void *args, std::uint64_t voidStarSize,
110 std::uint64_t resultOffset) override {
111 cudaq::info(
112 "BaseRemoteSimulatorQPU: Launch kernel named '{}' remote QPU {} "
113 "(simulator = {})",
114 name, qpu_id, m_simName);
115
116 cudaq::ExecutionContext *executionContextPtr =
117 getExecutionContextForMyThread();
118
119 if (executionContextPtr && executionContextPtr->name == "tracer") {
120 return;
121 }
122
123 // Default context for a 'fire-and-ignore' kernel launch; i.e., no context
124 // was set before launching the kernel. Use a static variable per thread to
125 // set up a single-shot execution context for this case.
126 static thread_local cudaq::ExecutionContext defaultContext("sample",
127 /*shots=*/1);
128 // This is a kernel invocation outside the CUDA-Q APIs (sample/observe).
129 const bool isDirectInvocation = !executionContextPtr;
130 cudaq::ExecutionContext &executionContext =
131 executionContextPtr ? *executionContextPtr : defaultContext;
132
133 // Populate the conditional feedback metadata if this is a direct
134 // invocation (not otherwise populated by cudaq::sample)
135 if (isDirectInvocation)
136 executionContext.hasConditionalsOnMeasureResults =
137 cudaq::kernelHasConditionalFeedback(name);
138
139 std::string errorMsg;
140 const bool requestOkay = m_client->sendRequest(
141 *m_mlirContext, executionContext, /*serializedCodeContext=*/nullptr,
142 /*vqe_gradient=*/nullptr, /*vqe_optimizer=*/nullptr, /*vqe_n_params=*/0,
143 m_simName, name, kernelFunc, args, voidStarSize, &errorMsg);
144 if (!requestOkay)
145 throw std::runtime_error("Failed to launch kernel. Error: " + errorMsg);
146 if (isDirectInvocation &&
147 !executionContext.invocationResultBuffer.empty()) {
148 if (executionContext.invocationResultBuffer.size() + resultOffset >
149 voidStarSize)
150 throw std::runtime_error(
151 "Unexpected result: return type size of " +
152 std::to_string(executionContext.invocationResultBuffer.size()) +
153 " bytes overflows the argument buffer.");
154 // Currently, we only support result buffer serialization on LittleEndian
155 // CPUs (x86, ARM, PPC64LE).
156 // Note: NVQC service will always be using LE. If
157 // the client (e.g., compiled from source) is built for big-endian, we
158 // will throw an error if result buffer data is returned.
159 if (llvm::sys::IsBigEndianHost)
160 throw std::runtime_error(
161 "Serializing the result buffer from a remote kernel invocation is "
162 "not supported for BigEndian CPU architectures.");
163
164 char *resultBuf = reinterpret_cast<char *>(args) + resultOffset;
165 // Copy the result data to the args buffer.
166 std::memcpy(resultBuf, executionContext.invocationResultBuffer.data(),
167 executionContext.invocationResultBuffer.size());
168 executionContext.invocationResultBuffer.clear();
169 }
170 }

225class BaseNvcfSimulatorQPU : public cudaq::BaseRemoteSimulatorQPU {
226public:
227 BaseNvcfSimulatorQPU() : BaseRemoteSimulatorQPU() {
228 m_client = cudaq::registry::get<cudaq::RemoteRuntimeClient>("NVCF");
229 }
230
231 // Encapsulates Nvcf configurations that we need.
232 // Empty strings mean no config available.
233 struct NvcfConfig {
234 std::string apiKey;
235 std::string functionId;
236 std::string versionId;
237 };
238
239 virtual void setTargetBackend(const std::string &backend) override {
240 auto parts = cudaq::split(backend, ';');
241 if (parts.size() % 2 != 0)
242 throw std::invalid_argument("Unexpected backend configuration string. "
243 "Expecting a ';'-separated key-value pairs.");
244 std::string apiKey, functionId, versionId, ngpus;
245
246 for (std::size_t i = 0; i < parts.size(); i += 2) {
247 if (parts[i] == "simulator")
248 m_simName = parts[i + 1];
249 // First, check if api key or function Id is provided as target options.
250 if (parts[i] == "function_id")
251 functionId = parts[i + 1];
252 if (parts[i] == "api_key")
253 apiKey = parts[i + 1];
254 if (parts[i] == "version_id")
255 versionId = parts[i + 1];
256 if (parts[i] == "ngpus")
257 ngpus = parts[i + 1];
258 }
259 // If none provided, look for them in environment variables or the config
260 // file.
261 const auto config = searchNvcfConfig();
262 if (apiKey.empty())
263 apiKey = config.apiKey;
264 if (functionId.empty())
265 functionId = config.functionId;
266 if (versionId.empty())
267 versionId = config.versionId;
268
269 // API key and function Id are required.
270 if (apiKey.empty())
271 throw std::runtime_error(
272 "Cannot find NVQC API key. Please refer to the documentation for "
273 "information about obtaining and using your NVQC API key.");
274
275 if (!apiKey.starts_with("nvapi-"))
276 std::runtime_error(
277 "An invalid NVQC API key is provided. Please check your settings.");
278 std::unordered_map<std::string, std::string> clientConfigs{
279 {"api-key", apiKey}};
280 if (!functionId.empty())
281 clientConfigs.emplace("function-id", functionId);
282 if (!versionId.empty())
283 clientConfigs.emplace("version-id", versionId);
284 if (!ngpus.empty())
285 clientConfigs.emplace("ngpus", ngpus);
286
287 m_client->setConfig(clientConfigs);
288 }
289
290 // The NVCF version of this function needs to dynamically fetch the remote
291 // capabilities from the currently deployed servers.
292 virtual RemoteCapabilities getRemoteCapabilities() const override {
293 return m_client->getRemoteCapabilities();
294 }
295
296protected:
297 // Helper to search NVQC config from environment variable or config file.
298 NvcfConfig searchNvcfConfig() {
299 NvcfConfig config;
300 // Search from environment variable
301 if (auto apiKey = std::getenv("NVQC_API_KEY"))
302 config.apiKey = std::string(apiKey);
303
304 if (auto funcIdEnv = std::getenv("NVQC_FUNCTION_ID"))
305 config.functionId = std::string(funcIdEnv);
306
307 if (auto versionIdEnv = std::getenv("NVQC_FUNCTION_VERSION_ID"))
308 config.versionId = std::string(versionIdEnv);
309
310 std::string nvqcConfig;
311 // Allow someone to tweak this with an environment variable
312 if (auto creds = std::getenv("CUDAQ_NVQC_CREDENTIALS"))
313 nvqcConfig = std::string(creds);
314 else
315 nvqcConfig = std::string(getenv("HOME")) + std::string("/.nvqc_config");
316 if (cudaq::fileExists(nvqcConfig)) {
317 std::ifstream stream(nvqcConfig);
318 std::string contents((std::istreambuf_iterator<char>(stream)),
319 std::istreambuf_iterator<char>());
320 std::vector<std::string> lines;
321 lines = cudaq::split(contents, '\n');
322 for (const std::string &l : lines) {
323 std::vector<std::string> keyAndValue = cudaq::split(l, ':');
324 if (keyAndValue.size() != 2)
325 throw std::runtime_error("Ill-formed configuration file (" +
326 nvqcConfig +
327 "). Key-value pairs must be in `<key> : "
328 "<value>` format. (One per line)");
329 cudaq::trim(keyAndValue[0]);
330 cudaq::trim(keyAndValue[1]);
331 if (config.apiKey.empty() &&
332 (keyAndValue[0] == "key" || keyAndValue[0] == "apikey"))
333 config.apiKey = keyAndValue[1];
334 if (config.functionId.empty() && (keyAndValue[0] == "function-id" ||
335 keyAndValue[0] == "Function ID"))
336 config.functionId = keyAndValue[1];
337 if (config.versionId.empty() &&
338 (keyAndValue[0] == "version-id" || keyAndValue[0] == "Version ID"))
339 config.versionId = keyAndValue[1];
340 }
341 }
342 return config;
343 }
344};
