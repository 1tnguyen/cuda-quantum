45template <typename ScalarType>
46class CusvState : public cudaq::SimulationState {
47private:
49 std::size_t size = 0;

55 bool ownsDevicePtr = true;

60 void checkAndSetDevice() const {
61 int dev = 0;
62 HANDLE_CUDA_ERROR(cudaGetDevice(&dev));
63 auto currentDevice = deviceFromPointer(devicePtr);
64 if (dev != currentDevice)
65 HANDLE_CUDA_ERROR(cudaSetDevice(currentDevice));
66 }
67
70 void extractValues(std::complex<ScalarType> *value, std::size_t start,
71 std::size_t end) const {
72 checkAndSetDevice();
73 HANDLE_CUDA_ERROR(cudaMemcpy(
74 value, reinterpret_cast<std::complex<ScalarType> *>(devicePtr) + start,
75 (end - start) * sizeof(std::complex<ScalarType>),
76 cudaMemcpyDeviceToHost));
77 }
78
80 bool isDevicePointer(void *ptr) const {
81 cudaPointerAttributes attributes;
82 HANDLE_CUDA_ERROR(cudaPointerGetAttributes(&attributes, ptr));
83 return attributes.type > 1;
84 }
85
87 int deviceFromPointer(void *ptr) const {
88 cudaPointerAttributes attributes;
89 HANDLE_CUDA_ERROR(cudaPointerGetAttributes(&attributes, ptr));
90 return attributes.device;
91 }
92
95 auto maybeCopyToDevice(std::size_t size, void *dataPtr) {
96 if (isDevicePointer(dataPtr))
97 return dataPtr;
98
99 std::complex<ScalarType> *ptr = nullptr;
100 HANDLE_CUDA_ERROR(
101 cudaMalloc((void **)&ptr, size * sizeof(std::complex<ScalarType>)));
102 HANDLE_CUDA_ERROR(cudaMemcpy(ptr, dataPtr,
103 size * sizeof(std::complex<ScalarType>),
104 cudaMemcpyHostToDevice));
105 return reinterpret_cast<void *>(ptr);
106 };
107
108public:
109 CusvState(std::size_t s, void *ptr) : size(s), devicePtr(ptr) {}
110 CusvState(std::size_t s, void *ptr, bool owns)
111 : size(s), devicePtr(ptr), ownsDevicePtr(owns) {}
112
114 std::size_t getNumQubits() const override { return std::log2(size); }
115
119 std::complex<double> overlap(const cudaq::SimulationState &other) override {
120 if (getTensor().extents != other.getTensor().extents)
121 throw std::runtime_error("[custatevec-state] overlap error - other state "
122 "dimension not equal to this state dimension.");
123
124 if (other.getPrecision() != getPrecision()) {
125 throw std::runtime_error(
126 "[custatevec-state] overlap error - precision mismatch.");
127 }
128
129 // It could be that the current set Device is not
130 // where the data resides.
131 int currentDev;
132 cudaGetDevice(&currentDev);
133 auto dataDev = deviceFromPointer(devicePtr);
134 if (currentDev != dataDev)
135 cudaSetDevice(dataDev);
136
137 // Make sure other is on GPU device already
138 if (isDevicePointer(other.getTensor().data)) {
139 if (deviceFromPointer(devicePtr) !=
140 deviceFromPointer(other.getTensor().data))
141 throw std::runtime_error(
142 "overlap requested for device pointers on separate GPU devices.");
143
144 auto cmplx = nvqir::innerProduct<ScalarType>(
145 devicePtr, other.getTensor().data, size, false);
146 return std::abs(std::complex<ScalarType>(cmplx.real, cmplx.imaginary));
147 } else {
148 // If we reach here, then we have to copy the data from host.
149 cudaq::info("[custatevec-state] overlap computation requested with a "
150 "state that is "
151 "in host memory. Host data will be copied to GPU.");
152
153 auto cmplx = nvqir::innerProduct<ScalarType>(
154 devicePtr, other.getTensor().data, size, true);
155 return std::abs(std::complex<ScalarType>(cmplx.real, cmplx.imaginary));
156 }
157 }
158
159 std::complex<double>
160 getAmplitude(const std::vector<int> &basisState) override {
161 if (getNumQubits() != basisState.size())
162 throw std::runtime_error(
163 fmt::format("[custatevec-state] getAmplitude with an invalid number "
164 "of bits in the "
165 "basis state: expected {}, provided {}.",
166 getNumQubits(), basisState.size()));
167 if (std::any_of(basisState.begin(), basisState.end(),
168 [](int x) { return x != 0 && x != 1; }))
169 throw std::runtime_error(
170 "[custatevec-state] getAmplitude with an invalid basis state: only "
171 "qubit state (0 or 1) is supported.");
172
173 // Convert the basis state to an index value
174 const std::size_t idx = std::accumulate(
175 std::make_reverse_iterator(basisState.end()),
176 std::make_reverse_iterator(basisState.begin()), 0ull,
177 [](std::size_t acc, int bit) { return (acc << 1) + bit; });
178 std::complex<ScalarType> value;
179 extractValues(&value, idx, idx + 1);
180 return {value.real(), value.imag()};
181 }

184 void dump(std::ostream &os) const override {
185 // get state data from device to print
186 std::vector<std::complex<ScalarType>> tmp(size);
187 HANDLE_CUDA_ERROR(cudaMemcpy(tmp.data(), devicePtr,
188 size * sizeof(std::complex<ScalarType>),
189 cudaMemcpyDeviceToHost));
190 for (auto &t : tmp)
191 os << t << "\n";
192 }

195 bool isDeviceData() const override { return true; }

198 const void *getDevicePointer() const { return devicePtr; }

201 precision getPrecision() const override {
202 if constexpr (std::is_same_v<ScalarType, float>)
203 return cudaq::SimulationState::precision::fp32;
204
205 return cudaq::SimulationState::precision::fp64;
206 }

208 std::unique_ptr<SimulationState>
209 createFromSizeAndPtr(std::size_t size, void *ptr, std::size_t type) override {
210 // If the data is provided as a pointer / size, then
211 // we assume we do not own it.
212 bool weOwnTheData = type < 2 ? true : false;
213 ptr = maybeCopyToDevice(size, ptr);
214 return std::make_unique<CusvState<ScalarType>>(size, ptr, weOwnTheData);
215 }

219 Tensor getTensor(std::size_t tensorIdx = 0) const override {
220 if (tensorIdx != 0)
221 throw std::runtime_error("[cusv-state] invalid tensor requested.");
222 return Tensor{devicePtr, std::vector<std::size_t>{size}, getPrecision()};
223 }
224
226 std::vector<Tensor> getTensors() const override { return {getTensor()}; }
227
229 std::size_t getNumTensors() const override { return 1; }
230
231 std::complex<double>
232 operator()(std::size_t tensorIdx,
233 const std::vector<std::size_t> &indices) override {
234 if (tensorIdx != 0)
235 throw std::runtime_error("[cusv-state] invalid tensor requested.");
236
237 if (indices.size() != 1)
238 throw std::runtime_error("[cusv-state] invalid element extraction.");
239
240 auto idx = indices[0];
241 std::complex<ScalarType> value;
242 extractValues(&value, idx, idx + 1);
243 return {value.real(), value.imag()};
244 }
245
247 void toHost(std::complex<double> *userData,
248 std::size_t numElements) const override {
249 // Must have the correct precision
250 if constexpr (std::is_same_v<ScalarType, float>)
251 throw std::runtime_error("simulation precision is FP32 but toHost "
252 "requested with FP64 host buffer.");
253
254 // Must have the correct number of elements.
255 if (numElements != size)
256 throw std::runtime_error("[custatevec-state] provided toHost pointer has "
257 "invalid number of elements specified.");
258
259 extractValues(reinterpret_cast<std::complex<ScalarType> *>(userData), 0,
260 size);
261 return;
262 }
263
265 void toHost(std::complex<float> *userData,
266 std::size_t numElements) const override {
267 // Must have the correct precision
268 if constexpr (std::is_same_v<ScalarType, double>)
269 throw std::runtime_error("simulation precision is FP64 but toHost "
270 "requested with FP32 host buffer.");
271
272 // Must have the correct number of elements.
273 if (numElements != size)
274 throw std::runtime_error("[custatevec-state] provided toHost pointer has "
275 "invalid number of elements specified.");
276
277 extractValues(reinterpret_cast<std::complex<ScalarType> *>(userData), 0,
278 size);
279 return;
280 }
281
283 void destroyState() override {
284 if (!ownsDevicePtr)
285 return;
286
287 int currentDev;
288 cudaGetDevice(&currentDev);
289 auto device = deviceFromPointer(devicePtr);
290 if (currentDev != device)
291 cudaSetDevice(device);
292
293 cudaGetDevice(&currentDev);
294 cudaq::info("custatevec-state destroying state vector handle (devicePtr "
295 "GPU = {}, currentDevice = {}).",
296 device, currentDev);
297
298 HANDLE_CUDA_ERROR(cudaFree(devicePtr));
299 }
