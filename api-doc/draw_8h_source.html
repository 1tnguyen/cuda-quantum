30template <typename KernelFunctor, typename... Args>
31cudaq::Trace traceFromKernel(KernelFunctor &&kernel, Args &&...args) {
32 // Get the platform.
33 auto &platform = cudaq::get_platform();
34
35 // This can only be done in simulation
36 if (!platform.is_simulator())
37 throw std::runtime_error("Cannot use draw on a physical QPU.");
38
39 // Create an execution context, indicate this is for tracing the execution
40 // path
41 ExecutionContext context("tracer");
42
43 // set the context, execute and then reset
44 platform.set_exec_ctx(&context);
45 kernel(args...);
46 platform.reset_exec_ctx();
47
48 return context.kernelTrace;
49}

128template <typename QuantumKernel, typename... Args>
129 requires std::invocable<QuantumKernel &, Args...>
130#else
131template <
132 typename QuantumKernel, typename... Args,
133 typename = std::enable_if_t<std::is_invocable_v<QuantumKernel, Args...>>>
134#endif
135std::string draw(std::string format, QuantumKernel &&kernel, Args &&...args) {
136 if (format == "ascii") {
137 return draw(kernel, std::forward<Args>(args)...);
138 } else if (format == "latex") {
139 return __internal__::getLaTeXString(
140 details::traceFromKernel(kernel, std::forward<Args>(args)...));
141 } else {
142 throw std::runtime_error(
143 "Invalid format. Supported formats are 'ascii' and 'latex'.");
144 }
145}
