41 kraus_op(std::vector<cudaq::complex> d) : data(d) {
42 auto nElements = d.size();
43 auto sqrtNEl = std::sqrt(nElements);
44 if (sqrtNEl * sqrtNEl != nElements)
45 throw std::runtime_error(
46 "Invalid number of elements to kraus_op. Must be square.");
47
48 nRows = (std::size_t)std::round(sqrtNEl);
49 nCols = nRows;
50 }
51
53 template <typename T>
54 kraus_op(std::initializer_list<T> &&initList)
55 : data(initList.begin(), initList.end()) {
56 auto nElements = initList.size();
57 auto sqrtNEl = std::sqrt(nElements);
58 if (sqrtNEl * sqrtNEl != nElements)
59 throw std::runtime_error(
60 "Invalid number of elements to kraus_op. Must be square.");
61
62 nRows = (std::size_t)std::round(sqrtNEl);
63 nCols = nRows;
64 }
65
67 kraus_op &operator=(const kraus_op &other) {
68 data = other.data;
69 return *this;
70 }
71
73 kraus_op adjoint() const {
74 std::size_t N = data.size();
75 std::vector<cudaq::complex> newData(N);
76 for (std::size_t i = 0; i < nRows; i++)
77 for (std::size_t j = 0; j < nCols; j++)
78 newData[i * nRows + j] = std::conj(data[j * nCols + i]);
79 return kraus_op(newData);
80 }

163 using KeyT = std::pair<std::string, std::vector<std::size_t>>;

166 struct KeyTHash {
167 template <class T, class U>
168 std::size_t operator()(const std::pair<T, U> &p) const {
169 auto hash = std::hash<T>{}(p.first);
170 hash ^= p.second.size();
171 for (auto &i : p.second) {
172 hash ^= i + 0x9e3779b9 + (hash << 6) + (hash >> 2);
173 }
174 return hash;
175 }
176 };

241class depolarization_channel : public kraus_channel {
242public:
243 depolarization_channel(const real probability) : kraus_channel() {
244 auto three = static_cast<real>(3.);
245 auto negOne = static_cast<real>(-1.);
246 std::vector<cudaq::complex> k0v{std::sqrt(1 - probability), 0, 0,
247 std::sqrt(1 - probability)},
248 k1v{0, std::sqrt(probability / three), std::sqrt(probability / three),
249 0},
250 k2v{0, cudaq::complex{0, negOne * std::sqrt(probability / three)},
251 cudaq::complex{0, std::sqrt(probability / three)}, 0},
252 k3v{std::sqrt(probability / three), 0, 0,
253 negOne * std::sqrt(probability / three)};
254 ops = {k0v, k1v, k2v, k3v};
255 validateCompleteness();
256 }
