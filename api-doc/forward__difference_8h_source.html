18public:

22 virtual std::unique_ptr<cudaq::gradient> clone() override {
23 auto newGrad = std::make_unique<forward_difference>(*this);
24 newGrad->step = this->step;

29 void compute(const std::vector<double> &x, std::vector<double> &dx,
30 const spin_op &h, double funcAtX) override {
31 auto tmpX = x;
32 for (std::size_t i = 0; i < x.size(); i++) {
33 // increase value to x_i + dx_i
34 tmpX[i] += step;
35 auto px = getExpectedValue(tmpX, h);
36 // return value back to x_i
37 tmpX[i] -= step;
38 dx[i] = (px - funcAtX) / step;
39 }
40 }

45 compute(const std::vector<double> &x,

51 // increase value to x_i + dx_i
