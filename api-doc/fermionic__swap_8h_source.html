18__qpu__ void global_phase(double theta, cudaq::qubit &q) {
19 // R1(theta) and Rz(theta) are equivalent upto a global phase.
20 // Hence, R1(phi) Rz(-phi) sequence results in a global phase of e^(i *
21 // phi/2).
22 // TODO: For hardware execution, where access to the state vector is
23 // impossible, hence global phase is irrelevant, there are a couple of
24 // optimization passes that we can do: (1) Remove this r1(angle) - rz (-angle)
25 // pattern at the global scope when we know that no control modifier has been
26 // applied. (2) Optimize the controlled gate decomposition (phase kickback on
27 // the control qubits).
28 r1(2.0 * theta, q);
29 rz(-2.0 * theta, q);
30}

44__qpu__ void fermionic_swap(double phi, cudaq::qubit &q0, cudaq::qubit &q1) {
45 h(q0);
46 h(q1);
47
48 x<cudaq::ctrl>(q0, q1);
49 rz(phi / 2.0, q1);
50 x<cudaq::ctrl>(q0, q1);
51
52 h(q0);
53 h(q1);
54
55 rx(M_PI_2, q0);
56 rx(M_PI_2, q1);
57
58 x<cudaq::ctrl>(q0, q1);
59 rz(phi / 2.0, q1);
60 x<cudaq::ctrl>(q0, q1);
61
62 rx(-M_PI_2, q0);
63 rx(-M_PI_2, q1);
64 rz(phi / 2.0, q0);
65 rz(phi / 2.0, q1);
66
67 // Global phase correction
68 global_phase(phi / 2.0, q0);
69}

75void fermionic_swap(KernelBuilder &kernel, cudaq::QuakeValue phi,
76 cudaq::QuakeValue q0, cudaq::QuakeValue q1) {
77 kernel.h(q0);
78 kernel.h(q1);
79
80 kernel.template x<cudaq::ctrl>(q0, q1);
81 kernel.rz(phi / 2.0, q1);
82 kernel.template x<cudaq::ctrl>(q0, q1);
83
84 kernel.h(q0);
85 kernel.h(q1);
86
87 kernel.rx(M_PI_2, q0);
88 kernel.rx(M_PI_2, q1);
89
90 kernel.template x<cudaq::ctrl>(q0, q1);
91 kernel.rz(phi / 2.0, q1);
92 kernel.template x<cudaq::ctrl>(q0, q1);
93
94 kernel.rx(-M_PI_2, q0);
95 kernel.rx(-M_PI_2, q1);
96 kernel.rz(phi / 2.0, q0);
97 kernel.rz(phi / 2.0, q1);
98
99 // Global phase correction
100 kernel.r1(phi, q0);
101 kernel.rz(-phi, q0);
102}
