143 Result<std::invoke_result_t<std::decay_t<KernelTy>, std::decay_t<Args>...>>>
144run(std::size_t shots, KernelTy &&f, Args &&...args) {
145 using resultTy = Result<
146 std::invoke_result_t<std::decay_t<KernelTy>, std::decay_t<Args>...>>;
147 if (shots < 1)
148 throw std::invalid_argument("The number of shots must be greater than 0.");
149 std::vector<resultTy> results;
150 results.reserve(shots);
151 for (std::size_t i = 0; i < shots; ++i) {
152 try {
153 results.emplace_back(resultTy(f(args...)));
154 } catch (std::exception &e) {
155 results.emplace_back(resultTy(e));
156 }
157 }
158 return results;
159}

213run_async(std::size_t qpu_id, std::size_t shots, KernelTy &&f, Args &&...args) {
214 using resultTy = Result<
215 std::invoke_result_t<std::decay_t<KernelTy>, std::decay_t<Args>...>>;
216 if (shots < 1)
217 throw std::invalid_argument("The number of shots must be greater than 0.");
218 auto &platform = cudaq::get_platform();
219 if (qpu_id >= platform.num_qpus())
220 throw std::invalid_argument(
221 "Provided qpu_id is invalid (must be <= to platform.num_qpus()).");
222
223 std::promise<std::vector<resultTy>> promise;
224 auto fut = promise.get_future();
225 // Wrapped it as a generic (returning void) function
226 QuantumTask wrapped = detail::make_copyable_function(
227 [p = std::move(promise), qpu_id, shots, &platform, &f,
228 args = std::make_tuple(std::forward<Args>(args)...)]() mutable {
229 std::vector<resultTy> results;
230 results.reserve(shots);
231 for (std::size_t i = 0; i < shots; ++i) {
232 try {
233 results.emplace_back(resultTy(std::apply(f, args)));
234 } catch (std::exception &e) {
235 results.emplace_back(resultTy(e));
236 }
237 }
238 p.set_value(std::move(results));
239 });
240
241 platform.enqueueAsyncTask(qpu_id, wrapped);
242 return fut;
243}
