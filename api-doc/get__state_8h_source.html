36template <typename KernelFunctor>
37state extractState(KernelFunctor &&kernel) {
38 // Get the platform.
39 auto &platform = cudaq::get_platform();
40
41 // This can only be done in simulation
42 if (!platform.is_simulator())
43 throw std::runtime_error("Cannot use get_state on a physical QPU.");
44 // Create an execution context, indicate this is for
45 // extracting the state representation
46 ExecutionContext context("extract-state");
47
48 // Perform the usual pattern set the context,
49 // execute and then reset
50 platform.set_exec_ctx(&context);
51 kernel();
52 platform.reset_exec_ctx();
53
54 // Return the state data. Since the ExecutionContext
55 // is done being used, we'll move the simulation state
56 // pointer to the state type. The state will retain
57 // value semantics, due to its tracking of this simulation
58 // data as a shared_ptr.
59 return state(context.simulationState.release());
60}

62template <typename KernelFunctor>
63auto runGetStateAsync(KernelFunctor &&wrappedKernel,
64 cudaq::quantum_platform &platform, std::size_t qpu_id) {
65 // This can only be done in simulation
66 if (!platform.is_simulator())
67 throw std::runtime_error("Cannot use get_state_async on a physical QPU.");
68
69 if (qpu_id >= platform.num_qpus())
70 throw std::invalid_argument(
71 "Provided qpu_id is invalid (must be <=to platform.num_qpus()).");
72
73 std::promise<state> promise;
74 auto f = promise.get_future();
75 // Wrapped it as a generic (returning void) function
76 QuantumTask wrapped = detail::make_copyable_function(
77 [p = std::move(promise), qpu_id, &platform,
78 func = std::forward<KernelFunctor>(wrappedKernel)]() mutable {
79 ExecutionContext context("extract-state");
80 // Indicate that this is an async exec
81 context.asyncExec = true;
82 // Set the platform and the qpu id.
83 platform.set_exec_ctx(&context, qpu_id);
84 platform.set_current_qpu(qpu_id);
85 func();
86 platform.reset_exec_ctx(qpu_id);
87 // Extract state data
88 p.set_value(state(context.simulationState.release()));
89 });
90
91 platform.enqueueAsyncTask(qpu_id, wrapped);
92 return f;
93}

98template <typename QuantumKernel, typename... Args>
99auto get_state(QuantumKernel &&kernel, Args &&...args) {
100#if defined(CUDAQ_REMOTE_SIM) && !defined(CUDAQ_LIBRARY_MODE)
101 // If this is a kernel that we cannot retrieve a name at runtime (C-type
102 // function), we cannot use lazy evaluation since the kernel name/quake code
103 // is not retrievable. This needs to be directed to the `altLaunchKernel`
104 // function, whereby the bridge has generated code to construct the kernel
105 // name at runtime.
106 if (cudaq::get_quake_by_name(cudaq::getKernelName(kernel), false).empty())
107 return details::extractState([&]() mutable {
108 cudaq::invokeKernel(std::forward<QuantumKernel>(kernel),
109 std::forward<Args>(args)...);
110 });
111
112 return state(new RemoteSimulationState(std::forward<QuantumKernel>(kernel),
113 std::forward<Args>(args)...));
114#else
115#if defined(CUDAQ_REMOTE_SIM)
116 // Kernel builder is MLIR-based kernel.
117 if constexpr (has_name<QuantumKernel>::value) {
118 return state(new RemoteSimulationState(std::forward<QuantumKernel>(kernel),
119 std::forward<Args>(args)...));
120 }
121#endif
122 return details::extractState([&]() mutable {
123 cudaq::invokeKernel(std::forward<QuantumKernel>(kernel),
124 std::forward<Args>(args)...);
125 });
126#endif
127}

140#if CUDAQ_USE_STD20
141template <typename QuantumKernel, typename... Args>
142 requires KernelCallValid<QuantumKernel, Args...>
143#else
144template <typename QuantumKernel, typename... Args,
145 typename = std::enable_if_t<
146 std::is_invocable_r_v<void, QuantumKernel, Args...>>>
147#endif
148async_state_result get_state_async(std::size_t qpu_id, QuantumKernel &&kernel,
149 Args &&...args) {
150 auto &platform = cudaq::get_platform();
151#if CUDAQ_USE_STD20
152 return details::runGetStateAsync(
153 [&kernel, ... args = std::forward<Args>(args)]() mutable {
154 cudaq::invokeKernel(std::forward<QuantumKernel>(kernel),
155 std::forward<Args>(args)...);
156 },
157 platform, qpu_id);
158#else
159 return details::runGetStateAsync(
160 [args = std::make_tuple(kernel, args...)]() mutable {
161 std::apply(cudaq::invokeKernel, args);
162 },
163 platform, qpu_id);
164#endif
165}
